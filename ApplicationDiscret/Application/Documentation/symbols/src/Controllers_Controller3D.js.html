 78 /**
 79  * @constructor
 80  * 
 81  * @param {Vector} dimension - Vector to define the dimension of the 3D space.
 82  */ 
 83 function Controller3D (dimension) {
 84 	/// parameter verification
 85 	if (!(dimension instanceof Vector)) {
 86 		console.error ("Controller3D.constructor: "
 87 				+ "given parameter is not a Vector");
 88 	}
 89 	
 90 	Controller.call (this);
 91 	
 92 	/**
 93 	 * {modelGen} The model which generates the surface.
 94 	 */
 95 	this.modelGen = new ModelGen (dimension);
 96 	
 97 	/**
 98 	 * {Function} A function that provides a frozen reference to the generatrix
 99 	 */
100 	this.getMeridian = null;
101 	 
102 	/**
103 	 * {Function} A function that provides a frozen reference to the directrix
104 	 */
105 	this.getCurveRevolution = null;
106 };

111 /**
112  * Function that start the generation of the surface from the meridian and the
113  * curbe of revolution.
114  * 
115  * @param {int} mode - wether the surface should be generated using an 
116  * incrementalor a brute force method
117  * 
118  * @return {void}
119  * @throws {String} Controller3D.generate: The meridian and/or the curve revolution 
120 			are not functions.
121  */
122 Controller3D.prototype.generate = function (mode) {
123 	/// verification
124 	if (! checkType ([this.getMeridian, this.getCurveRevolution], Function,
125 		Function))
126 	{
127 		throw "Controller3D.generate: The meridian and/or the curve revolution "
128 			+ "are not functions."
129 		// FIXME trouver une meilleur phrase pour cette exception
130 	}
131 	
132 	var meridian = this.getMeridian ();
133 	var curveRevolution = this.getCurveRevolution ();
134 	this.modelGen.generate (meridian, curveRevolution, mode);
135 };

164 /**
165  * @param {(Number | Vector)} x - The X coordinate (if it is a number) or all
166  * coordinates (if it is a Vector) of the voxel.
167  * @param {Number} [y] - The Y coordinate of the voxel.
168  * @param {Number} [z] - The Z coordinate of the voxel.
169  * 
170  * @return {Voxel} The voxel at the x, y, z coordinates.
171  * @throws {String} Controller3D.getVoxel.ErrorLengthArguments - Wrong number of arguments.
172  * @throws {String} Controller3D.getVoxel.ErrorNotAVector - There is only one argument but it's not a vector.
173  * @throws {String} Controller3D.getVoxel.ErrorNotANumber - There are three arguments but at least one of them is not a number.
174  */
175 Controller3D.prototype.getVoxel = function (x, y, z) {
176 	switch (arguments.length) {
177 		case 1 :
178 			if (!(x instanceof Vector)) {
179 				throw "Controller3D.getVoxel.ErrorNotAVector";
180 			}
181 			return this.modelGen.getVoxel (x);
182 			break;
183 		case 3 :
184 			if (! checkType (arguments, "number", "number", "number")) {
185 				throw "Controller3D.getVoxel.ErrorNotANumber";
186 			}
187 			return this.modelGen.getVoxel (new Vector (x, y, z));
188 			break;
189 		default :
190 			throw "Controller3D.getVoxel.ErrorLengthArguments";
191 	}
192 };

196 /**
197  * TODO
198  * 
199  * @param {(Number | Vector)} x - The X coordinate (if it is a number) or all coordinates
200  * (if it is a Vector) of the voxel.
201  * @param {Number} [y] - The Y coordinate of the voxel.
202  * @param {Number} [z] - The Z coordinate of the voxel.
203  * 
204  * @throws {String} Controller3D.hasVoxel.ErrorLengthArguments - Wrong number of arguments.
205  * @throws {String} Controller3D.hasVoxel.ErrorNotAVector - There is only one argument but it's not a vector.
206  * @throws {String} Controller3D.hasVoxel.ErrorNotANumber - There are three arguments but at least one of them is not a number.
207  * 
208  * @return {boolean} True if the voxel is a voxel of the surface, false
209  * otherwise.
210  */
211 Controller3D.prototype.hasVoxel = function (x, y, z) {
212 	switch (arguments.length) {
213 		case 1 :
214 			if (!(x instanceof Vector)){
215 				throw "Controller3D.hasVoxel.ErrorNotAVector";
216 			}
217 			return this.modelGen.getVoxel (x) !== null;
218 			break;
219 		case 3 :
220 			if (typeof x !== "number"
221 				|| typeof y !== "number"
222 				|| typeof z !== "number"){
223 				throw "Controller3D.hasVoxel.ErrorNotANumber";
224 			}
225 			return this.modelGen.getVoxel (new Vector (x, y, z)) !== null;
226 			break;
227 		default :
228 			throw "Controller3D.hasVoxel.ErrorLengthArguments";
229 	}
230 };
