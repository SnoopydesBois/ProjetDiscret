 77 /**
 78  * @constructor
 79  * Constructs a viewer for a parameter list.
 80  * 
 81  * @param {String} idContainer - Id of the associated controller. If the first
 82  * character is a '#', it is ignored.
 83  */
 84 function ParameterViewer (idContainer) {
 85 
 86 	/**
 87 	 * {HTMLUlElement} The list which contains the ui parameters.
 88 	 */
 89 	this.container = null;
 90 	if (idContainer.charAt (0) !== '#')
 91 		this.container = $('#' + idContainer);
 92 	else
 93 		this.container = $(idContainer);
 94 
 95 	/**
 96 	 * {Controller2D} The associated controller. Used to get all the parameters 
 97 	 * of a curve.
 98 	 * @see {@link displayParameter}
 99 	 */
100 	this.controller = null;
101 };

129 /**
130  * Fills the list of parameters (empty if before). 
131  *
132  * @param {Function} draw - A function to redraw the curve. Used to add an event
133  * on the input to redraw the curve when modified.
134  * @param {Function} getRange - A function to get the range of a parameter. 
135  *
136  * @return {void}
137  */
138 ParameterViewer.prototype.displayParameter = function (draw, getRange) {
139 
140 	this.container.empty ();
141 
142 	var listParameter = this.controller.getAllParameters ();
143 	var id = this.container.attr ("id");
144 	var that = this;
145 
146 	var range;
147 	for (var param in listParameter) {
148 		range = getRange (param);
149 
150 		var li = $("<li></li>");
151 
152 		li.append ("<p class='titleParam'>  " + param + "</p>");
153 		var input = $("<input type='range' name='" + param
154 			+ "' id='" + id + param
155 			+ "' class='rangeParam' value='" + listParameter[param]
156 			+ "' min='" + range.getMin ()
157 			+ "' max='" + range.getMax ()
158 			+ "'/>"
159 		);
160 
161 		input.change (function () {
162 			that.controller.setParameter ($(this).attr ("name"),
163 				parseInt ($(this).val ())
164 			);
165 			draw ();
166 		});
167 		li.append (input);
168 		this.container.append (li);
169 	} // end for each param
170 };
