 50 /**
 51  * @enum
 52  * Enumeration of axis.
 53  * /!\ CAUTION: values aren't number ! Each axis is an object which contain a
 54  * 'colorOffset' attribute.
 55  */
 56 var AxisEnum = {
 57 	/// values
 58 	X : {colorOffset :  0.0},
 59 	Y : {colorOffset : -0.1},
 60 	Z : {colorOffset :  0.1},
 61 
 62 	/// properties
 63 	properties : {
 64 		0 : {name : "X"},
 65 		1 : {name : "Y"},
 66 		2 : {name : "Z"}
 67 	},
 68 
 69 	/// functions
 70 	/**
 71 	 * Convert number to AxisEnum.
 72 	 *
 73 	 * @param {Number} axis - A number.
 74 	 *
 75 	 * @return {AxisEnum} The matching axis to the number.
 76 	 */
 77 	toAxis : function (axis) {
 78 		switch (axis) {
 79 			case 0: return AxisEnum.X; break;
 80 			case 1: return AxisEnum.Y; break;
 81 			case 2: return AxisEnum.Z; break;
 82 			default : return null; break;
 83 		}
 84 	}
 85 };
 86 
 87 
 88 
 89 if (Object.freeze){
 90 	AxisEnum = Object.freeze (AxisEnum);
 91 }
