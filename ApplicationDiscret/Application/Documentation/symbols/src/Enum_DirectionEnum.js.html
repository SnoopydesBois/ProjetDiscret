 49 /**
 50  * @enum
 51  * Enumeration of Directions.
 52  * x, y and z indicates how much me move along this axis;
 53  * oppose is the value of the opposite face;
 54  * size indicate the number of directions.
 55  */
 56 var DirectionEnum = {
 57 	/// values
 58 	TOP    : 0,
 59 	BOTTOM : 1,
 60 	RIGHT  : 2,
 61 	LEFT   : 3,
 62 	FRONT  : 4,
 63 	BACK   : 5,
 64 	ALL    : 6,
 65 	NONE   : 7,
 66 
 67 	/// properties
 68 	properties : {
 69 		0 : {x :  0, y :  0, z :  1, oppose : 1, axis : AxisEnum.Z, name : "TOP"},
 70 		1 : {x :  0, y :  0, z : -1, oppose : 0, axis : AxisEnum.Z, name : "BOTTOM"},
 71 		2 : {x :  1, y :  0, z :  0, oppose : 3, axis : AxisEnum.X, name : "RIGHT"},
 72 		3 : {x : -1, y :  0, z :  0, oppose : 2, axis : AxisEnum.X, name : "LEFT"},
 73 		4 : {x :  0, y : -1, z :  0, oppose : 5, axis : AxisEnum.Y, name : "FRONT"},
 74 		5 : {x :  0, y :  1, z :  0, oppose : 4, axis : AxisEnum.Y, name : "BACK"},
 75 		6 : {x :  0, y :  0, z :  0, oppose : 7, axis : null, name : "ALL"},
 76 		7 : {x :  0, y :  0, z :  0, oppose : 6, axis : null, name : "NONE"}
 77 	},
 78 	size : 6,
 79 
 80 	/// methods
 81 	toString : function (direction) {
 82 		if (0 <= direction && direction <= 7){
 83 			return DirectionEnum.properties[direction].name;
 84 		} else{
 85 			return undefined;
 86 		}
 87 	}
 88 };
 89 
 90 
 91 
 92 if (Object.freeze){
 93 	DirectionEnum = Object.freeze (DirectionEnum);
 94 }
