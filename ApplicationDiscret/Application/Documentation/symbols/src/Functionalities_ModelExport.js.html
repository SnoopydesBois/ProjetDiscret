217 /**
218  * Exports the 3D canvas to X3D. Proposes to save an X3D file representing the
219  * surface by boxes.
220  * 
221  * @param {Surface} surface - The surface to export to X3D file.
222  * 
223  * @return {void}
224  * @throws {String} The parameter is not of type Surface.
225  */
226 ModelExport.prototype.exportX3D = function (surface) {
227 	if (! (surface instanceof Surface)) {
228 		throw "ModelExport.exportX3D.ErrorBadParameterType";
229 	}
230 	var x3D = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" + 
231 					"\t<!DOCTYPE X3D PUBLIC \"ISO//Web3D//DTD X3D 3.2//EN\"\n" +
232 					  "\t\t\"http://www.web3d.org/specifications/" +
233 					  "x3d-3.2.dtd\">\n\n" +
234 
235 					"\t<X3D profile=\"Immersive\" version=\"3.2\"\n" +
236 						 "\t\txmlns:xsd=\"http://www.w3.org/2001/XMLSchema" +
237 						 "-instance\"\n" +
238 						 "\t\txsd:noNamespaceSchemaLocation=\"http://www.web3d"+
239 						 ".org/specifications/x3d-3.2.xsd\">\n" +
240 						 "\t<head></head>\n" +
241 					"\t<Scene>\n" ;
242 	
243 	var dimension = surface.getDimension ();
244 	
245 	// Gets the current connexity.
246 	var connexity = parseInt ($("#connexityChoice").val ());
247 	
248 	// Creates the voxels in the X3D file
249 	for (var x = 0; x < dimension.x; x++) {
250 		for (var y = 0; y < dimension.y; y++) {
251 			for (var z = 0; z < dimension.z; z++) {
252 				var voxel = surface.getVoxel (x, z, y)
253 				if (voxel != null && voxel.isVisible (connexity)) {
254 					x3D += "\t<Transform translation=\""+ (x - dimension.x / 2) 
255 									+ " " 
256 									+ (y - dimension.y / 2) 
257 									+ " " 
258 									+ (z - dimension.z / 2)  
259 									+ "\">\n" +
260 								"\t\t<Shape>\n" + 
261 									"\t\t\t<Box size=\"" + (1.0 - dimension.x/2) 
262 										 + " " 
263 										 + (1.0 - dimension.y / 2) 
264 										 + " " 
265 										 + (1.0 - dimension.z / 2)
266 										 + "\"/>\n"+
267 								"\t\t</Shape>\n" +
268 							"\t</Transform>\n";
269 				} // end if
270 			} // end for z
271 		} // end for y
272 	} // end for x
273 
274 	x3D += "\t</Scene>\n" +
275 		"</X3D>";
276 		
277 	var blob = new Blob([x3D], {type: "text/xml"});
278 
279 	saveAs(blob, "surfaceX3d.x3d");
280 };

306 /**
307  * Use of the saveSvgAsPng function from saveSvgAsPng library.
308  * 
309  * @param {String} id - The name of the div containing the svg to save.
310  * @param {String} name - The name of the file to save.
311  * 
312  * @return {void}
313  * @throws {String} If a parameter doesn't have the excepted type.
314  */
315 ModelExport.prototype.getImg2DData = function (id, name) {
316 	if (!checkType (arguments, "string", "string")) {
317 		throw "ModelExport.getImg2DData.ErrorBadParameterType";
318 	}
319 	// If there is no svg, document.querySelector return null. 
320 	// It happens when we are working on the canvas (hand free drawing)
321 	var svgElement = document.querySelector("#" + id + '>svg');
322 	
323 	if (svgElement == null) { // Canvas case
324 		var canvas = document.querySelector ("#" + id);
325 		var width = canvas.width;
326 		var height = canvas.height
327 		var context = canvas.getContext ("2d");
328 		
329 		var imageData = context.getImageData (0, 0, width, height);
330 				
331 		canvas.toBlob (function (blob) {
332 			saveAs (blob, "generatrix.png");
333 		});
334 	}
335 	else { // SVG case
336 		saveSvgAsPng (svgElement, name);
337 	}
338 };

362 /**
363  * Creates 2D canvas from 3D canvas.
364  * 
365  * @param {String} id - The name of the div containing the svg to save
366  * @param {SurfaceViewer} surfaceView - The surfaceView containing the objects
367  * to render.
368  * 
369  * @return {String} The url to the canvas img.
370  * @throws {String} If a parameter doesn't have the excepted type.
371  */
372 ModelExport.prototype.getImg3DData = function (id, surfaceView) {
373 	if (!((arguments, "string", SurfaceViewer))) {
374 		throw "ModelExport.getImg3DData.ErrorBadParameterType";
375 	}
376 
377 	var width = $("#" + id).width (), height = $("#" + id).height ();
378 	
379 	var pixels = [];
380 	pixels = surfaceView.getImgData (width, height);
381 		
382 	var canvas = document.createElement ('canvas');
383 	canvas.width = width;
384 	canvas.height = height;
385 	var context = canvas.getContext ("2d");
386 	
387 	var imageData = context.createImageData (width, height);
388 	imageData.data.set (pixels);
389 	context.putImageData (imageData, 0, 0);
390 	
391 	return canvas.toDataURL ();
392 };

396 /**
397  * Exports the current surface to binary STL format
398  * 
399  * @param {SurfaceRenderer} renderer - The surface renderer.
400  * 
401  * @return {void}
402  * @throws {String} If a parameter doesn't have the excepted type.
403  */
404 ModelExport.prototype.exportSTL = function (renderer) {
405 	if (!(renderer instanceof SurfaceRenderer)) {
406 		throw "ModelExport.exportSTL.ErrorBadParameterType";
407 	}
408 	
409 	var isLittleIndian = true; // STL files assume little endian
410 	// see https://en.wikipedia.org/wiki/STL_%28file_format%29
411 
412 	// Retrieving the coordinates
413 	var vertexBuffer = [];
414 	
415 	var indicesBuffer = [];
416 	var connexity = parseInt ($("#connexityChoice").val ());
417 	if (connexity == ConnexityEnum.C26) {
418 		connexity = ConnexityEnum.C18;
419 	}
420 	renderer.prepareSTL (connexity, indicesBuffer, vertexBuffer);
421 	
422 	var nbTriangles = indicesBuffer.length / 3;
423 	// Creation of the buffer containing the data for the stl
424 	
425 	/* UINT8[80] – Header
426 	   UINT32 – Number of triangles
427 	 */
428 	var headerOffset = 84;
429 	/*  REAL32[3] – Normal vector
430 		REAL32[3] – Vertex 1
431 		REAL32[3] – Vertex 2
432 		REAL32[3] – Vertex 3
433 		UINT16 – Attribute byte count
434 	 */
435 	var sizeTriangles = 50 * nbTriangles;
436 	
437 	var buffer = new ArrayBuffer (headerOffset + sizeTriangles);
438 
439 	// Creation of a view to write data in the buffer
440 	var dataview = new DataView (buffer);
441 	// Offset used when writing data into buffer
442 	var offset = 80; // For the STL header 
443 		
444 		
445     dataview.setUint32 (offset, nbTriangles, isLittleIndian);
446 	offset += 4;
447 	
448 	for (var i = 0; i < indicesBuffer.length; i+=3) {
449 		dataview.setFloat32(offset, 0.0, isLittleIndian);
450 		offset += 4; // size of float in bytes
451 		dataview.setFloat32(offset, 0.0, isLittleIndian);
452 		offset += 4;
453 		dataview.setFloat32(offset, 0.0, isLittleIndian);
454 		offset += 4;
455 		for (var p =  0; p < 3; p++) {
456 			for (var c = 0; c < 3; c++) {
457 				dataview.setFloat32(offset, vertexBuffer[
458 						indicesBuffer[i + p] * 3 + c
459 					], isLittleIndian);
460 				offset += 4;
461 			}
462 		}
463 		offset += 2; // ignore byte count attribute
464 	}
465 	
466 	var blob = new Blob ([dataview], {type: 'application/octet-binary'});
467     
468     // FileSaver.js defines `saveAs` for saving files out of the browser
469     saveAs (blob, "surfaceSTL.stl");
470 };

484 	if (!(meridianController instanceof Controller2DMeridian)) {
485 		throw "ModelExport.saveGeneratrix.ErrorBadParameterType";
486 	}
487 	
488 	var curve = meridianController.getActiveCurve ();
489 	if (curve instanceof ExplicitCurve) {
490 		this.saveExplicitCurve (meridianController);
491 	}
492 	else if (curve instanceof DrawnCurve) {
493 		this.saveDrawnCurve (meridianController);
494 	}
495 };

499 /**

501  * 
502  * @param {Controller2D} curveController - The controller of the curve to

504  * 
505  * @return {void}
506  * @throws {String} If the parameter is not of type Controller2D.

526 				/* ====== */

528 				'\t<CurveType>\n' +
529 					'\t\t<Class>ImplicitCurve</Class>\n' +
530 				'\t</CurveType>\n' +
531 				'\t<Equation>'+  curveController.getEquationNoParameter()

636 					'\t</CurveType>\n' +
637 					'\t<xCoords>';
