 61 /**
 62  * Initializes all the application (controller, event, interface).
 63  *
 64  * @return {void}
 65  */
 66 Application.prototype.initAppli = function () {
 67 	/// Application initialization
 68 	this.verifNavigator ();
 69 	this.initControllers ();
 70 
 71 	this.meridianController.setActive ("Line");
 72 	this.revolController.setActive ("Ellipse");
 73 
 74 	this.meridianParameters.setController (this.meridianController);
 75 	this.meridianParameters.displayParameter (this.drawMeridian,
 76 		this.getRangeMeridian);
 77 
 78 	this.revolutionParameters.setController (this.revolController);
 79 	this.revolutionParameters.displayParameter (this.drawRevolution,
 80 		this.getRangeRevolution);
 81 
 82 
 83 	/// Interface initialization
 84 
 85 	// Double slider creation
 86 	var dimX = document.getElementById ("dimx").value;
 87 	var dimY = document.getElementById ("dimy").value;
 88 	var dimZ = document.getElementById ("dimz").value;
 89 	this.createSlider ("#slider-rangeX", 0, dimX);
 90 	this.createSlider ("#slider-rangeY", 0, dimY);
 91 	this.createSlider ("#slider-rangeZ", 0, dimZ);
 92 	
 93 	$(".buttonGroup").buttonset ();
 94 	$(".buttonGroup button")
 95 		.removeClass("ui-button ui-widget ui-state-default ui-button-text-only")
 96 		.removeAttr ("role")
 97 		.unbind ();
 98 	this.changeDimension ();
 99 	this.resizeInterface ();
100 	this.initWindowEvent ();
101 	this.showDefaultMessage ();
102 	this.changeMeridianMode ();
103 };

107 /**
108  * Initializes meridian and revolution curve controller. Initializes the functions used to draw curves
109  * and get range.
110  *
111  * @return {void}
112  */
113 Application.prototype.initControllers = function () {
114 	/// add curve for meridian
115 	this.meridianController.addCurve ("Line", Line);
116 	this.meridianController.addCurve ("Sinusoid", Sinusoid);
117 
118 	/// add curve for revolution
119 	this.revolController.addCurve ("Ellipse", Ellipse);
120 	this.revolController.addCurve ("Heart", Heart);
121 	this.revolController.addCurve ("Lemniscate", Lemniscate);
122 	this.revolController.addCurve ("Lissajous", Lissajous);
123 	this.revolController.addCurve ("Quadrifolium", Quadrifolium);
124 	this.revolController.addCurve ("Knot", Knot);
125 	this.revolController.addCurve ("Scarabeus", Scarabeus);
126 	this.revolController.addCurve ("Hyperbol", Hyperbol);
127 	this.revolController.addCurve ("Folium", Folium);
128 	this.revolController.addCurve ("Hexafolium", Hexafolium);
129 	this.revolController.addCurve ("Curve1", Curve1);
130 	this.revolController.addCurve ("Curve2", Curve2);
131 	this.revolController.addCurve ("Curve3", Curve3);
132 
133 	/// bind generation with active curve
134 	this.surfaceController.setGetCurveRevolution (
135 		this.revolController.getActiveCurve.bind (this.revolController)
136 	);
137 	this.surfaceController.setGetMeridian (
138 		this.meridianController.getActiveCurve.bind (this.meridianController)
139 	);
140 
141 	this.drawMeridian = this.meridianView.draw.bind (this.meridianView);
142 	this.drawRevolution = this.revolView.draw.bind (this.revolView);
143 
144 	this.getRangeMeridian = this.meridianController.getParametersRange
145 		.bind (this.meridianController);
146 	this.getRangeRevolution = this.revolController.getParametersRange
147 		.bind (this.revolController);
148 };

163 /**
164  * Transforms a div element into a double slider (with jQuery).
165  * 
166  * @param {(HTMLDivElement | String)} id - The div element or its id (with the first character being a '#').
167  * @param {int} min - Minimum value of the slider.
168  * @param {int} max - Maximum value of the slider.
169  * 
170  * @return {void}
171  */
172 Application.prototype.createSlider = function (id, min, max) {
173 	// Slider creation
174 	$(id).slider ({
175 		range : true,
176 		min : min,
177 		max : max,
178 		values: [min, max],
179 		slide: function (event, ui) {
180 			if ((ui.values[0]) >= ui.values[1]) {
181 				return false;
182 			}
183 			$("#amountMin" + $(id).attr ("name")).val (ui.values[0]);
184 			$("#amountMax" + $(id).attr ("name")).val (ui.values[1]);
185 		},
186 		stop : function(event, ui){
187 			appli.show (true)
188 		}
189 	});
190 
191 	// Text input initialisation
192 	$("#amountMin" + $(id).attr ("name")).val ($(id).slider ('values', 0));
193 	$("#amountMax" + $(id).attr ("name")).val ($(id).slider ('values', 1));
194 };

198 /**
199  * Checks the navigator. If it is not "Mozilla" (for Firefox and Chrome), displays
200  * a pop up.
201  * 
202  * @return {void}
203  */
204 Application.prototype.verifNavigator = function () {
205 	if (navigator.appCodeName != "Mozilla")
206 		// erase style="display: none"
207 		document.getElementById ("notMozillaPopUp").style = ""; 
208 };
