249 /**
250  * Loads the meridian curve from an XML file.
251  * 
252  * @param {Event} event - The event.
253  * 
254  * @return {void}
255  */
256 Application.prototype.loadMeridian = function (event) {
257 	var tmppath = URL.createObjectURL (event.target.files[0]);
258 
259 	var request = new XMLHttpRequest ();
260 	request.open ("GET", tmppath, false);
261 	request.send ();
262 
263 	var xml = request.responseXML;
264 
265 	var typeCurve = xml.getElementsByTagName ("Class")[0].textContent
266         .toString ();
267 	var equation, xPoints, yPoints, xMin, xMax, yMin, yMax;
268 
269 	equation = xml.getElementsByTagName ("Equation");
270 	if (equation.length == 0) {
271 		// Same length for both of the lists
272 		xPoints = xml.getElementsByTagName ("xCoords")[0].textContent
273             .toString ().split (" ");
274 		yPoints = xml.getElementsByTagName ("yCoords")[0].textContent
275             .toString ().split (" ");
276 	}
277 	else {
278 		// Same length since each parameters got a name and a value
279 		xMin = parseFloat (xml.getElementsByTagName ("xMin")[0].textContent
280             .toString ());
281 		xMax = parseFloat (xml.getElementsByTagName ("xMax")[0].textContent
282             .toString ());
283 		yMin = parseFloat (xml.getElementsByTagName ("yMin")[0].textContent
284             .toString ());
285 		yMax = parseFloat (xml.getElementsByTagName ("yMax")[0].textContent
286             .toString ());
287 	}
288 
289 	if(typeCurve === "DrawnCurve"){
290 		$("#meridianFreeHand").prop ("checked", true);
291 		$("#meridianPrimitive").prop ("checked", false);
292 		$("#meridianFormula").prop ("checked", false);
293 		this.changeMeridianMode ()
294 		this.meridianController.newCurve ();
295 		var drawnCurve = this.meridianController.getActiveCurve ();
296 		for (var i = 0; i < xPoints.length; ++i) {
297 			drawnCurve.addPoint (xPoints[i], yPoints[i]);
298 		}
299 	}
300 	else if (typeCurve === "ExplicitCurve" || typeCurve === "ImplicitCurve") {
301 		this.meridianController.setActive (
302             equation[0].textContent.toString (),
303             EquationTypeEnum.EXPLICIT
304         );
305 
306 		this.meridianController.setXRange (new Range (xMin, xMax));
307 		this.meridianController.setYRange (new Range (yMin, yMax));
308 	}
309 
310 	this.meridianParameters.displayParameter (
311 		this.drawMeridian,
312 		this.getRangeMeridian
313 	);
314 	this.meridianView.draw ();
315 };

319 /**
320  * Loads the revolution curve from an XML file.
321  * 
322  * @param {Event} event - The event.
323  * 
324  * @return {void}
325  */
326 Application.prototype.loadRevolution = function (event) {
327 	var tmppath = URL.createObjectURL (event.target.files[0]);
328 
329 	var request = new XMLHttpRequest ();
330 	request.open ("GET", tmppath, false);
331 	request.send ();
332 
333 	var xml = request.responseXML;
334 
335 	var typeCurve = xml.getElementsByTagName ("Class")[0].textContent
336         .toString ();
337 
338 	if (typeCurve !== "ImplicitCurve") {
339 		return;
340 	}
341 
342 	var equation, listNameParameters, listValueParameters, xPoints, yPoints,
343         xMin, xMax, yMin, yMax;
344 	equation = xml.getElementsByTagName ("Equation");
345 	// Same length since each parameters got a name and a value
346 	listNameParameters = xml.getElementsByTagName ("Name");
347 	listValueParameters = xml.getElementsByTagName ("Value");
348 
349 	xMin = parseFloat (xml.getElementsByTagName ("xMin")[0].textContent
350         .toString ());
351 	xMax = parseFloat (xml.getElementsByTagName ("xMax")[0].textContent
352         .toString ());
353 	yMin = parseFloat (xml.getElementsByTagName ("yMin")[0].textContent
354         .toString ());
355 	yMax = parseFloat (xml.getElementsByTagName ("yMax")[0].textContent
356         .toString ());
357 
358 	this.revolController.setActive (
359         equation[0].textContent.toString (),
360         EquationTypeEnum.IMPLICIT
361     );
362 
363 	for (var i = 0; i < listNameParameters.length; ++i) {
364 		this.revolController.setParameter (
365             listNameParameters[i].textContent.toString (),
366             parseFloat (listValueParameters[i].textContent.toString ())
367         );
368 	}
369 
370 	this.revolController.setXRange (new Range (xMin, xMax));
371 	this.revolController.setYRange (new Range (yMin, yMax));
372 
373 	this.revolutionParameters.displayParameter (
374 		this.drawRevolution,
375 		this.getRangeRevolution
376 	);
377 	this.revolView.draw ();
378 };

395 /**
396  * Moves the X slider to show the highlight X slice. Resets Y and Z sliders.
397  *
398  * @return {void}
399  */
400 Application.prototype.showXSlice = function () {
401 	var s = this.surfaceView.getSurfaceRenderer ();
402 	if (s && s.highlightX != -1) {
403 		var $slider = $("#slider-rangeX");
404 		var x = s.highlightX + 1;
405 		var minX = $slider.slider ('values', 0);
406 		if (minX > x) {
407 			minX = x - 1;
408 			x = $slider.slider ("values", 1);
409 		}
410 		var slider = $slider.slider (
411 			"option", "values", [minX, x]);
412 		this.resetSlider ("slider-rangeY", false);
413 		this.resetSlider ("slider-rangeZ", false);
414 		$("#amountMinX").val (minX);
415 		$("#amountMaxX").val (x);
416 		this.show (true);
417 	}
418 };

476 /**
477  * Hides every highlighted slice and redraws the scene. If there is not any surface,
478  * does nothing.
479  *
480  * @return {void}
481  */
482 Application.prototype.hideSlice = function () {
483 	var s = this.surfaceView.getSurfaceRenderer ();
484 	if (s) {
485 		s.highlightX = -1;
486 		s.highlightY = -1;
487 		s.highlightZ = -1;
488 		this.show (true);
489 	}
490 };
