366 /**
367  * Translates each element (HTMLtext, title and href).
368  * 
369  * @return {void}
370  */
371 Application.prototype.translateAll = function () {
372 	this.activeIndexLang = (this.activeIndexLang + 1)
373 		% Application.languages.length;
374 	activeLang = Application.languages[this.activeIndexLang];
375 	
376 	/// translate HTMLtext
377 	$('.tr').each (function (i) {
378 		$(this).text (aLangKeys[activeLang][$(this).attr ('id')]);			
379 	});		
380 
381 	/// translate title
382 	$('.trtitle').each (function (i) {
383 		$(this).attr ("title", 
384 			translateTitle[activeLang][$(this).attr ('id')]);			
385 	});
386 	
387 	/// translate a element
388 	$('.atr').each (function (i) {
389 		$(this).attr ("href", translateHref[activeLang][$(this).attr ('id')]);			
390 	});
391 };
