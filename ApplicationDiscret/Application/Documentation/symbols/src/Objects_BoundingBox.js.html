 77 /**
 78  * @constructor 
 79  * Creates a bounding box.
 80  * 
 81  * @param {Vector} dimension - The dimensions of the 3D space.
 82  * @param {WebGLRenderingContext} glContext - The gl context (used by the
 83  * shader).
 84  */
 85 function BoundingBox (dimension, glContext) {
 86 	GenericStructure.call (this, "boundingBox", new DefaultShader (glContext));
 87 		
 88 	/**
 89 	 * {Vector} The dimension of the 3D space in voxel.
 90 	 */
 91 	this.dimension = dimension;
 92 	
 93 	/**
 94 	 * @inheritdoc
 95 	 * An user cannot select the box.
 96 	 */
 97 	this.isPickable = false;
 98 };

118  * Sets the dimension of the generate surface.
119  * 
120  * @param {(Vector | Number[3] | Number} dimension - The dimensions of the 3D
121  * space (@see {@link Vector}).
122  * 
123  * @return {void}
124  * @throws {String} The parameters are not of expected type
125  */
126 BoundingBox.prototype.setDimension = function (dimension) {
127 	/// parameter verification 
128 	if (! checkType (arguments, [Vector, Array, "number"])) {
129 		throw "BoundingBox.setDimension: bad type of parameter";
130 	}
131 	
132 	/// sets value
133 	this.dimension = new Vector (dimension);
134 	this.prepared = false;

145 /**
146  * @override
147  * 
148  * @param {WebGLRenderingContext} glContext - The gl context.
149  */
150 BoundingBox.prototype.prepare = function (glContext) {
151 	var halfDimBox = (new Vector (this.dimension)).mul (
152 		1.01 / Math.max (this.dimension.x, this.dimension.y, this.dimension.z));
153 	
154 	/// Vertex and color buffers
155 	var vertexBuffer = [
156 		[-halfDimBox.x, -halfDimBox.y, -halfDimBox.z],
157 		[-halfDimBox.x,  halfDimBox.y, -halfDimBox.z],
158 		[ halfDimBox.x, -halfDimBox.y, -halfDimBox.z],
159 		[ halfDimBox.x,  halfDimBox.y, -halfDimBox.z],
160 		[-halfDimBox.x, -halfDimBox.y,  halfDimBox.z],
161 		[-halfDimBox.x,  halfDimBox.y,  halfDimBox.z],
162 		[ halfDimBox.x, -halfDimBox.y,  halfDimBox.z],
163 		[ halfDimBox.x,  halfDimBox.y,  halfDimBox.z],
164 	];
165 
166 	var vertexBufferLength = vertexBuffer.length; // 8
167 	// Color of each vertices, lines are white
168 	var color = [0.7, 0.7, 0.7, 1.0];
169 
170 
171 	/// Vertex Buffer
172 	this.glVertexBuffer = glContext.createBuffer ();
173 	this.glVertexBuffer.numItems = vertexBufferLength; 
174 	
175 	var data = [];
176 	for (var vertice = 0; vertice < vertexBufferLength; ++vertice) {
177 		this.addAPoint (data, vertexBuffer[vertice]);
178 		this.addAColor (data, color);
179 	}
180 	glContext.bindBuffer (glContext.ARRAY_BUFFER, this.glVertexBuffer); 
181 	glContext.bufferData (glContext.ARRAY_BUFFER, new Float32Array (data), 
182 		glContext.STATIC_DRAW);
183 	
184 	
185 	/// Indices Buffer
186 	/*    5-------7   Y Z
187 	 *   /|      /|   |/
188 	 *  / |     / |   *--X
189 	 * 1-------3  |
190 	 * |  4----|--6  back face (4, 5, 7, 6)
191 	 * | /     | /
192 	 * |/      |/
193 	 * 0-------2  front face (0, 1, 3, 2)
194 	 */
195 	var indicesBuffer = [
196 		0, 2,  2, 3,  3, 1,  1, 0, // Front
197 		0, 4,  2, 6,  3, 7,  1, 5, // Middle
198 		4, 6,  6, 7,  7, 5,  5, 4  // Back
199 		
200 	];
201 	this.glIndiciesBuffer = glContext.createBuffer();
202 	this.glIndiciesBuffer.numItems = indicesBuffer.length;
203 	glContext.bindBuffer(glContext.ELEMENT_ARRAY_BUFFER, this.glIndiciesBuffer);
204 	glContext.bufferData (
205 		glContext.ELEMENT_ARRAY_BUFFER,
206 		new Uint16Array (indicesBuffer), 
207 		glContext.STATIC_DRAW
208 	);
209 	
210 	/// Finished, tells it
211 	this.prepared = true;
212 };

216 /**
217  * @override
218  * 
219  * @param {WebGLRenderingContext} glContext - The gl context.
220  * 
221  * @return {void}
222  */
223 BoundingBox.prototype.draw = function (glContext) {
224 	/// Parameter verification
225 	if (!(glContext instanceof WebGLRenderingContext)) {
226 		console.error ("BoundingBox.draw: glContext is not a "
227 			+ "WebGLRenderingContext");
228 		return;
229 	}
230 	
231 	/// Buffers verification
232 	if (this.glVertexBuffer === undefined 
233 		|| this.glIndiciesBuffer === undefined)
234 	{
235 		console.error ("BoundingBox.draw: prepare the box BEFORE drawing it !");
236 		return;
237 	}
238 		
239 	/// Set shader parameters
240 	this.shader.setRenderingMode (RenderingModeEnum.PICKING);
241 	// Let's the shader prepare its attributes
242 	this.shader.setAttributes (glContext, this.glVertexBuffer,
243 		globalParam.perspectiveView ? 1 : 0);
244 	
245 	// Let's render !
246 	glContext.bindBuffer(glContext.ELEMENT_ARRAY_BUFFER, this.glIndiciesBuffer);
247 	glContext.drawElements (
248 		glContext.LINES, 
249 		this.glIndiciesBuffer.numItems, 
250 		glContext.UNSIGNED_SHORT,
251 		0
252 	);
253 };
