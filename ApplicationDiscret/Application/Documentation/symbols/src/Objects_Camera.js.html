103 /**
104  * @constructor Perspective camera class.
105  *
106  * @param {Vector} eyePos - Position of the camera.
107  * @param {Vector} centerPos - Position of the center where the camera look at.
108  * @param {Vector} up - Perpendicular vector (going up) to centerPos vector.
109  * @param {int} width - Width of the camera.
110  * @param {int} height - Height of the camera.
111  * @param {float} fov - Field of view of the camera.
112  * @param {float} near - Nearest point of the camera.
113  * @param {float} far - The farthest point of the camera.
114  */
115 function Camera (eyePos, centerPos, up, width, height, fov, near, far) {
116 
117 	/**
118 	 * {Vector} Position of the camera.
119 	 */
120 	this.eyePos = eyePos;
121 
122 	/**
123 	 * {Vector} Point where the camera look at.
124 	 */
125 	this.centerPos = centerPos;
126 
127 	/**
128 	 * {Vector} Perpendicular vector (going up) to centerPos vector.
129 	 */
130 	this.up = up;
131 
132 	/**
133 	 * {int} Width of the camera. Should be the canvas width.
134 	 */
135 	this.width = width;
136 
137 	/**
138 	 * {int} Height of the camera. Should be the canvas height.
139 	 */
140 	this.height = height;
141 
142 	/**
143 	 * {float} Field of view of the camera. Angle in degrees.
144 	 */
145 	this.fov = fov || 45;
146 
147 	/**
148 	 * {float} Nearest point of the camera.
149 	 */
150 	this.near = near || 0.1;
151 
152 	/**
153 	 * {float} Farthest point of the camera.
154 	 */
155 	this.far = far || 1000.0;
156 
157 	/**
158 	 * {Matrix} View matrix
159 	 */
160 	this.viewMatrix = null;
161 
162 	/**
163 	 * {Matrix} Perspective projection matrix
164 	 */
165 	this.perspectiveProjectionMatrix = null;
166 
167 	/**
168 	 * {Matrix} orthographic projection matrix
169 	 */
170 	this.orthographicProjectionMatrix = null;
171 
172 	/**
173 	 * {float} Allows to modify the zoom.
174 	 */
175 	this.constProjection = 2.5;
176 
177 	/// Matrix calcul
178 	this.computeMatrices ();
179 };

252 Camera.prototype.setFov = function (aFov) {
253 	if (typeof aFov == "number") {
254 		this.fov = aFov;
255 		this.computeMatrices ();

272 /**
273  * Sets the orthographic zoom.
274  *
275  * @param {float} [nb] - Zoom.
276  *
277  * @return {void}
278  */
279 Camera.prototype.setProjection = function (nb) {
280 	if (typeof nb == "undefined") {
281 		this.constProjection = 2.5;
282 		this.computeMatrices ();
283 	}
284 	else if (0 < nb) {
285 		this.constProjection = nb;
286 		this.computeMatrices ();
287 	}
288 };

298 /**
299  * Computes view and projections matrices.
300  *
301  * @return {void}
302  */
303 Camera.prototype.computeMatrices = function () {
304 	// View matrix
305 	this.viewMatrix = new Matrix (this.eyePos, this.centerPos, this.up);
306 
307 	// Perspective matrix
308 	this.perspectiveProjectionMatrix = new Matrix (
309 		this.fov * Math.PI / 180,
310 		this.width / this.height,
311 		this.near,
312 		this.far
313 	);
314 
315 	// Orthographic matrix
316 	var factor = Math.min (this.height, this.width);
317 	this.orthographicProjectionMatrix = new Matrix (
318 		-this.constProjection * this.width / factor,
319 		 this.constProjection * this.width / factor,
320 		-this.constProjection * this.height / factor,
321 		 this.constProjection * this.height / factor,
322 		this.near,
323 		this.far
324 	);
325 };

329 /**
330  * Camera zoom.
331  *
332  * @param {Number} factor - Zoom factor.
333  *
334  * @return {void}
335  */
336 Camera.prototype.zoom = function (factor) {
337 	// orthographic zoom
338 	this.setProjection (this.getProjection () + factor);
339 	// perspective zoom
340 	var normPos = new Vector (this.getPosition ())
341 		.sub (this.getLookAtPosition ())
342 		.normalize ();
343 	this.eyePos.x += normPos.x * factor;
344 	this.eyePos.y += normPos.y * factor;
345 	this.eyePos.z += normPos.z * factor;
346 	this.computeMatrices ();
347 };
348 
349 
350 //==============================================================================
351 /**
352  * Camera zoom in.
353  *
354  * @return {void}
355  */
356 Camera.prototype.zoomIn = function () {
357 	var factor = -(0.1 + this.constProjection * 0.05)
358 	if ((this.eyePos.x - this.centerPos.x)
359 		* (this.eyePos.x + factor - this.centerPos.x)
360 		+ (this.eyePos.y - this.centerPos.y)
361 		* (this.eyePos.y + factor - this.centerPos.y)
362 		+ (this.eyePos.z - this.centerPos.z)
363 		* (this.eyePos.z + factor - this.centerPos.z) >= 0)
364 	{
365 		this.zoom (factor);
366 	}
367 };
368 
369 
370 //==============================================================================
371 /**
372  * Camera zoom out.
373  *
374  * @return {void}
375  */
376 Camera.prototype.zoomOut = function () {
377 	if (this.constProjection < 24)
378 		this.zoom (0.1 + this.constProjection * 0.05);
379 };
