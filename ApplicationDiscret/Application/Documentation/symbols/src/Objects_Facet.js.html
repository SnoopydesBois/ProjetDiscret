 62 /**
 63  * @classdesc Wrapper used to get the position and the direction of a voxel's 
 64  * face
 65  */
 66 Facet.prototype.constructor = Facet;
 67 
 68 
 69 
 70 //##############################################################################
 71 //	Constructor
 72 //##############################################################################
 73 
 74 
 75 
 76 /**
 77  * @constructor
 78  * 
 79  * @param {Vector} cubeCoor - Vector to define the position of the cube.
 80  * @param {DirectionEnum} dir - The face of the cube.
 81  */
 82 function Facet (cubeCoor, dir) {
 83 	if (! checkType (arguments, Vector, "number")) {
 84 		console.error ("Facet.constructor: bad type(s) of parameter(s)");
 85 	}
 86 
 87 	/**
 88 	 * {Vector} The coordinates in the universe.
 89 	 */
 90 	this.cubeCoor = cubeCoor;
 91 
 92 	/**
 93 	 * {DirectionEnum} The direction of the face (ALL for a cube).
 94 	 */
 95 	this.direction = dir;
 96 };
 97 
 98 
 99 
100 //##############################################################################
101 //	Accessors and Mutators
102 //##############################################################################
103 
104 
105 
106 /**
107  * @return {Vector} The position.
108  */
109 Facet.prototype.getCube = function () {
110 	return this.cubeCoor;
111 };
112 
113 
114 //==============================================================================
115 /**
116  * @return {DirectionEnum} The direction of the face.
117  */
118 Facet.prototype.getDirection = function () {
119 	return this.direction;
120 };
121 
122 
123 
124 //##############################################################################
125 //	Other methods
126 //##############################################################################
127 
128 
129 /**
130  * Tests the equality between to Facet (i.e. same coordinates and same
131  * direction).
132  * 
133  * @param {Facet} face - An other facet.
134  * 
135  * @return {boolean} True if 'this' and 'face' are the same, false otherwise.
136  */
137 Facet.prototype.egale = function (face) {
138 	if (face) {
139 		return (this.cubeCoor.m[0] == face.cubeCoor.m[0]
140 			&& this.cubeCoor.m[1] == face.cubeCoor.m[1]
141 			&& this.cubeCoor.m[2] == face.cubeCoor.m[2]
142 			&& this.direction == face.direction);
143 	}
144 	else{
145 		return false;
146 	}
147 };
148 
149 
150 //==============================================================================
151 /**
152  * @return {String} A string representing the facet.
153  */
154 Facet.prototype.toString = function () {
155 	return "position: " + this.cubeCoor.toString() + ", direction: "
156 		+ DirectionEnum.toString (this.direction);
157 };
