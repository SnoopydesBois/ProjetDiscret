 23 /**
 24  * @extends Curve
 25  * @classdesc A curve defined by an implicit equation
 26  */
 27 ImplicitCurve.prototype = new FormulaCurve;
 28 ImplicitCurve.prototype.constructor = ImplicitCurve;
 29 
 30 
 31 //==============================================================================
 32 /**
 33  * @constructor {Equation} the equation of the curve
 34  */
 35 function ImplicitCurve(equation) {
 36 	FormulaCurve.call(this, equation);
 37 };
 38 
 39 
 40 //==============================================================================
 41 /**
 42  * Computes the points of the curve based on its equation
 43  * 
 44  * @param {Range} ranX - The x-axis range on which to compute the points
 45  * @param {Range} ranX - The y-axis range on which to compute the points
 46  * 
 47  * @return {Point[][]} An array composed of list of points to represent the curve
 48  * (for example, 1/x will create two list of points)
 49  * @throw {String} The equation is not defined or the parameters are not of type Range
 50  */
 51 ImplicitCurve.prototype.computePoints = function (ranX, ranY) {
 52 	if (! ranX instanceof Range || ! ranY instanceof Range) {
 53 		throw "ImplicitCurve.computePoints.ErrorNotARange";
 54 	}
 55 	else if (!this.equation.check) {
 56 		throw "ImplicitCurve.computePoints.ErrorEquationNotDefined";
 57 	}
 58 	
 59 	var result = [];
 60 	
 61 	var points = [];
 62 	
 63 	var xMin = ranX.getMin();
 64 	var xMax = ranX.getMax();
 65 	var yMin = ranY.getMin();
 66 	var yMax = ranY.getMax();
 67 	
 68 	var step = this.computeStep(ranX, ranY);
 69 	
 70 	for(var x = xMin; x <= xMax; x += step){	
 71 		for (var y = yMin; y <= yMax; y += step){
 72 			var compute = this.equation.compute([x,y]);
 73 			/* 
 74 			 * If the point on the circle is within the display range,
 75 			 * we add that point to the current connex part of the curve
 76 			 */
 77 			if(compute > -0.5 && compute < 0.5){
 78 				points.push(new Point(x,y));
 79 			}
 80 			/* 
 81 			 * Else, we push into the result array the current connex curve computed
 82 			 * and we start a new connex part.
 83 			 */
 84 			else if(points.length > 0){
 85 				result.push(points.slice(0));
 86 				points = [];
 87 			}
 88 		}		
 89 	}
 90 	
 91 	/* 
 92 	 * Pushing into result the last connex part computed of the curve
 93 	 */
 94 	if(points.length > 0){
 95 		result.push(points.slice(0));
 96 	}
 97 	return result;
 98 };
