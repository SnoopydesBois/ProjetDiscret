118 /**
119  * @constructor
120  * Depending on the number of argument to the constructor, call for 
121  * different matrix construction.
122  * 	0 : Default constructor : 4x4 matrix.
123  * 	1 : Copy constructor : copy of the matrix in argument.
124  * 	3 : Creation of a lookAt Matrix for a camera.
125  *  4 : Creation of a Perspective Matrix.
126  * 	6 : Creation of an Orthogonal Matrix.
127  * 	default : Error.
128  * 
129  * @return {Matrix} the new Matrix.
130  */
131 function Matrix () {
132 	switch (arguments.length) {
133 	case 0: 
134 		return this.defaultConstructor ();
135 	case 1: 
136 		return this.copyConstructor (arguments[0]);
137 	case 3:
138 		return this.createLookAt (arguments[0], arguments[1], arguments[2]);
139 	case 4:
140 		return this.createPerspective (
141 			arguments[0], arguments[1], 
142 			arguments[2], arguments[3]
143 		);
144 	case 6: 
145 		return this.createOrthographic (
146 			arguments[0], arguments[1],
147 			arguments[2], arguments[3],
148 			arguments[4], arguments[5]
149 		);
150 	default:
151 		throw "Bad Matrix constructor call";
152 	};
153 	return this;
154 };

193 /**
194  * Creates a lookAt Matrix.
195  * 
196  * @param {Vector} eye - The eye position.
197  * @param {Vector} center - The position where the camera look at.
198  * @param {Vector} up - The up vector.
199  * 
200  * @return {Matrix} The lookAt Matrix.
201  * @throw {String} The parameter are not of expected types
202  */
203 Matrix.prototype.createLookAt = function (eye, center, up) {
204 	if (! checkType (arguments, Vector, Vector, Vector))
205 		throw "Matrix.createLookAt: bad parameter" ;
206 	
207 	var dst = new Vector (center).sub (eye).normalize (); // Z
208 	var right = dst.cross (up).normalize (); // X
209 	var upCrossed = right.cross (dst).normalize (); // Y
210 
211 	this.m = new MAT_ARRAY_TYPE (16);
212 	// First line
213 	this.m[0] =   right.x;
214 	this.m[1] =   upCrossed.x;
215 	this.m[2] = - dst.x;
216 	this.m[3] =   0.0; 
217 	// Second line
218 	this.m[4] =   right.y;
219 	this.m[5] =   upCrossed.y; 
220 	this.m[6] = - dst.y; 
221 	this.m[7] =   0.0;
222 	// Third line
223 	this.m[8]  =   right.z;
224 	this.m[9]  =   upCrossed.z;
225 	this.m[10] = - dst.z;
226 	this.m[11] =   0.0;
227 	// Last line
228 	this.m[12] = - right.dot (eye);
229 	this.m[13] = - upCrossed.dot (eye);
230 	this.m[14] =   dst.dot (eye);
231 	this.m[15] =   1.0; 
232 	
233 	return this;
234 };

238 /**
239  * Creates a perspective view matrix.
240  * @see https://developer.mozilla.org/fr/docs/Web/API/WebGL_API/WebGL_model_view_projection
241  * 
242  * @param {float} fov - Field of view in radian.
243  * @param {float} ratio - The aspect ratio of the scene (width divided by
244  * the height).
245  * @param {float} near - Nearest point of the camera.
246  * @param {float} far - Farest point of the camera.
247  * 
248  * @return {Matrix} A perspective view matrix.
249  * @throws {String} One parameter is not of type number.
250  */
251 Matrix.prototype.createPerspective = function (fov, ratio, near, far) {
252 	/// parameters verification
253 	if (! checkType (arguments, "number", "number", "number", "number")) {
254 		throw "Matrix.createPerspective: one of parameters is not a number";
255 	}
256 	
257 	/// Fills the matrix
258 	this.m = new MAT_ARRAY_TYPE (16);
259 	var f = 1.0 / Math.tan (fov / 2);
260 	var rangeInv = 1 / (near - far);
261 	
262 	// First line
263 	this.m[0] = f / ratio;
264 	this.m[1] = 0.0;
265 	this.m[2] = 0.0;
266 	this.m[3] = 0.0;
267 	// Second line
268 	this.m[4] = 0.0;
269 	this.m[5] = f;
270 	this.m[6] = 0.0;
271 	this.m[7] = 0.0;
272 	// Third line
273 	this.m[8]  = 0.0;
274 	this.m[9]  = 0.0;
275 	this.m[10] = (near + far) * rangeInv;
276 	this.m[11] = -1.0;
277 	// Last line
278 	this.m[12] = 0.0;
279 	this.m[13] = 0.0;
280 	this.m[14] = near * far * rangeInv * 2.0;
281 	this.m[15] = 0.0;
282 	
283 	return this;
284 };

288 /**
289  * Creates an orthographic view matrix.
290  * 
291  * @param {float} left - X coordinate of the minimum corner.
292  * @param {float} right - X coordinate of the maximum corner.
293  * @param {float} bottom - Z coordinate of the minimum corner.
294  * @param {float} top - Z coordinate of the maximum corner.
295  * @param {float} near - Y coordinate of the minimum corner.
296  * @param {float} far - Y coordinate of the maximum corner.
297  * 
298  * @return {Matrix} An orthographic view matrix.
299  * @throw {String} A parameter is not of type number
300  */
301 Matrix.prototype.createOrthographic = function (left, right,
302 		bottom, top, near, far)
303 {
304 	/// parameters verification
305 	if (! checkType (arguments, "number", "number", "number", "number", 
306 		"number", "number"))
307 	{
308 		throw "Matrix.createOrthographic: one of parameters is not a number";
309 	}
310 	
311 	/// Fills the matrix
312 	this.m = new MAT_ARRAY_TYPE (16);
313 	var lr = 1.0 / (left - right),
314 		bt = 1.0 / (bottom - top),
315 		nf = 1.0 / (near - far);
316 	
317 	// First line
318 	this.m[0] = -2.0 * lr;
319 	this.m[1] = 0.0;
320 	this.m[2] = 0.0;
321 	this.m[3] = 0.0;
322 	
323 	// Second line
324 	this.m[4] = 0.0;
325 	this.m[5] = -2.0 * bt;
326 	this.m[6] = 0.0;
327 	this.m[7] = 0.0;
328 	
329 	// Third line
330 	this.m[8] = 0.0;
331 	this.m[9] = 0.0;
332 	this.m[10] = 2.0 * nf;
333 	this.m[11] = 0.0;
334 	
335 	// Last line
336 	this.m[12] = (left + right) * Math.min (lr, bt);
337 	this.m[13] = (top + bottom) * Math.min (lr, bt);
338 	this.m[14] = (far + near) * nf;
339 	this.m[15] = 1.0;
340 	
341 	return this;
342 };

358 Matrix.prototype.get = function (i, j) {
359 	if (i < 0 || i >= 4) 
360 		throw "Matrix getter: bad line value (" + i + ")";
361 	if (j < 0 || j >= 4) 
362 		throw "Matrix getter: bad column value (" + j + ")";

435 /**
436  * Transforms a matrix to a normal matrix ...
437  * 
438  * @return {Matrix} The new matrix.
439  */
440 Matrix.prototype.toNormal = function () {
441 	// sets last column to 0
442 	this.m[3] = this.m[7] = this.m[11] = 0.0;
443 	// sets last line to 0
444 	this.m[12] = this.m[13] = this.m[14] = this.m[15] = 0.0;
445 	
446 	// computes the 3x3 invert, and transpose it ...
447 	var a = this.m[0]; var b = this.m[4]; var c = this.m[8];
448 	var d = this.m[1]; var e = this.m[5]; var f = this.m[9];
449 	var g = this.m[2]; var h = this.m[6]; var i = this.m[10];
450 	
451 	var invDet = 1.0 / (a * (e * i - h * f) 
452 			+ d * (c * h - i * b) 
453 			+ g * (b * f - e * c)
454 		);
455 	// computes directly the transpose matrix
456 	this.m[0] = (e * i - f * h) * invDet;
457 	this.m[1] = (c * h - b * i) * invDet;
458 	this.m[2] = (b * f - c * e) * invDet;
459 	
460 	this.m[4] = (f * g - d * i) * invDet;
461 	this.m[5] = (a * i - c * g) * invDet;
462 	this.m[6] = (c * d - a * f) * invDet;
463 	
464 	this.m[8] = (d * h - e * g) * invDet;
465 	this.m[9] = (b * g - a * h) * invDet;
466 	this.m[10]= (a * e - b * d) * invDet;
467 	
468 	// and returns
469 	return this;
470 };

474 /**
475  * @param {Vector} v - A vector.
476  * 
477  * @return {Vector} A new vector, this vector is the result of the
478  * multiplication of the matrix with the parameter (and a coeff).
479  * @throw {String} The parameter is not of type Vector
480  */
481 Matrix.prototype.apply = function (v) {
482 	if (!v instanceof Vector) {
483 		throw("Matrix.apply: bad parameter");
484 	}
485 	var inv = 1.0 / this.m[15]; // do not forget!
486 	var x = v.m[0] * inv;
487 	var y = v.m[1] * inv;
488 	var z = v.m[2] * inv;
489 	return new Vector (
490 		this.m[0] * x + this.m[4] * y + this.m[8] * z + this.m[12] * inv,
491 		this.m[1] * x + this.m[5] * y + this.m[9] * z + this.m[13] * inv,
492 		this.m[2] * x + this.m[6] * y + this.m[10] * z + this.m[14] * inv);
493 };

513  * @param {Matrix} that - The matrix to compare with.
514  * 
515  * @return {Matrix} True if the matrix are equals, false otherwise.
516  * @throw {String} The parameter is not of type Matrix or the comparison 
517  * coefficient is not finite
518  */
519 Matrix.prototype.equals = function (that) {
520 	
521 	// Allows to compare 2 matrix with a coefficient
522 	var epsilon = (arguments.length == 2) ? arguments[1] : 0; 
523 	
524 	if ((!that instanceof Matrix) || (!isFinite (epsilon))) {
525 		throw("Matrix equals: bad parameter");
526 	}
527 	
528 	// Comparison
529 	for (var i = 0; i < 16; ++i) {
530 		if (Math.abs (this.m[i] - that.m[i]) > epsilon) {
531 			return false;
532 		}
533 	}
534 	return true;
535 };

562  * Multiplication of two matrices (*=).
563  * 
564  * @param {Matrix} that - The matrix to multiply with.
565  * 
566  * @return {Matrix} The matrix result of the multiplication.
567  * @throw {String} The parameter is not of type Matrix
568  */
569 Matrix.prototype.mul = function (that) {
570 	if (!(that instanceof Matrix)) {
571 		throw "Matrix multiplication: given parameter is not a Matrix" ;
572 	}
573 	var t = new MAT_ARRAY_TYPE (this.m);
574 	
575 	for (var i = 0; i < 4; ++i) {
576 		for (var j = 0; j < 4; ++j) {
577 			var v = 0;
578 			for (var k = 0; k < 4; ++k) {
579 				v += t[i + 4 * k] * that.get (k, j);
580 			}
581 			this.set (i, j, v);
582 		}
583 	}
584 	// mandatory: let the following line survive!
585 	return this;
586 };

591  * This becomes the transposition of this.
592  * 
593  * @return {Matrix} The matrix transposed.
594  */
595 Matrix.prototype.transpose = function () {
596 	for (var l = 0; l < 4; ++l) {
597 		for (var c = l + 1; c < 4; ++c) {
598 			var v = this.get (l, c);
599 			this.set (l, c, this.get (c, l));
600 			this.set (c, l, v);
601 		}
602 	}
603 	return this;
604 };

608 /**
609  * @return {Matrix} the matrix inverted.
610  */
611 Matrix.prototype.invert = function () {
612 	var a00 = this.m[0],  a01 = this.m[1],  a02 = this.m[2],  a03 = this.m[3];
613 	var a10 = this.m[4],  a11 = this.m[5],  a12 = this.m[6],  a13 = this.m[7];
614 	var a20 = this.m[8],  a21 = this.m[9],  a22 = this.m[10], a23 = this.m[11];
615 	var a30 = this.m[12], a31 = this.m[13], a32 = this.m[14], a33 = this.m[15];
616 	
617 	var b00 = a00 * a11 - a01 * a10;
618 	var b01 = a00 * a12 - a02 * a10;
619 	var b02 = a00 * a13 - a03 * a10;
620 	var b03 = a01 * a12 - a02 * a11;
621 	var b04 = a01 * a13 - a03 * a11;
622 	var b05 = a02 * a13 - a03 * a12;
623 	var b06 = a20 * a31 - a21 * a30;
624 	var b07 = a20 * a32 - a22 * a30;
625 	var b08 = a20 * a33 - a23 * a30;
626 	var b09 = a21 * a32 - a22 * a31;
627 	var b10 = a21 * a33 - a23 * a31;
628 	var b11 = a22 * a33 - a23 * a32;
629 	
630 	var invDet = 1 / (b00 * b11 - b01 * b10 + b02 * b09 + b03 * b08 
631 		- b04 * b07 + b05 * b06);
632 	
633 	this.m[0]  = ( a11 * b11 - a12 * b10 + a13 * b09) * invDet;
634 	this.m[1]  = (-a01 * b11 + a02 * b10 - a03 * b09) * invDet;
635 	this.m[2]  = ( a31 * b05 - a32 * b04 + a33 * b03) * invDet;
636 	this.m[3]  = (-a21 * b05 + a22 * b04 - a23 * b03) * invDet;
637 	this.m[4]  = (-a10 * b11 + a12 * b08 - a13 * b07) * invDet;
638 	this.m[5]  = ( a00 * b11 - a02 * b08 + a03 * b07) * invDet;
639 	this.m[6]  = (-a30 * b05 + a32 * b02 - a33 * b01) * invDet;
640 	this.m[7]  = ( a20 * b05 - a22 * b02 + a23 * b01) * invDet;
641 	this.m[8]  = ( a10 * b10 - a11 * b08 + a13 * b06) * invDet;
642 	this.m[9]  = (-a00 * b10 + a01 * b08 - a03 * b06) * invDet;
643 	this.m[10] = ( a30 * b04 - a31 * b02 + a33 * b00) * invDet;
644 	this.m[11] = (-a20 * b04 + a21 * b02 - a23 * b00) * invDet;
645 	this.m[12] = (-a10 * b09 + a11 * b07 - a12 * b06) * invDet;
646 	this.m[13] = ( a00 * b09 - a01 * b07 + a02 * b06) * invDet;
647 	this.m[14] = (-a30 * b03 + a31 * b01 - a32 * b00) * invDet;
648 	this.m[15] = ( a20 * b03 - a21 * b01 + a22 * b00) * invDet;
649 	
650 	return this;
651 };

686 	}
687

692 	}

699  * @param {float} angleRad - the angle of rotation (in rad).
700  * 
701  * @return {Matrix} this=RotateX(angeRad)*this 
702  */
703 Matrix.prototype.rotateX = function (angleRad) {
704 	var c = Math.cos (angleRad); 
705 	var s = Math.sin (angleRad);
706 	
707 	// only the second and third lines are modified
708 	for (var i = 0; i < 4; ++i) {
709 		var a1 = this.get (1, i);
710 		var a2 = this.get (2, i);
711 		this.set (1, i, c * a1 - s * a2);
712 		this.set (2, i, s * a1 + c * a2);
713 	}
714 
715 	return this;
716 };

763 /**
764  * @param {float} angleRad - The angle of rotation (in rad).
765  * @param {Vector} axe - Axe of the rotation.
766  * 
767  * @return {Matrix} The rotated matrix.
768  */
769 Matrix.prototype.rotate = function (angleRad, axe) {
770 	// see https://fr.wikipedia.org/wiki/Matrice_de_rotation
771 	var c = Math.cos (angleRad); 
772 	var s = Math.sin (angleRad);
773 	var that = new Matrix ();
774 	var n = new Vector (axe).normalize ();
775 	
776 	// first line
777 	that.m[0] = n.x * n.x * (1 - c) + c;
778 	that.m[1] = n.x * n.y * (1 - c) - n.z * s;
779 	that.m[2] = n.x * n.z * (1 - c) + n.y * s;
780 	// second line
781 	that.m[4] = n.y * n.x * (1 - c) + n.z * s;
782 	that.m[5] = n.y * n.y * (1 - c) + c;
783 	that.m[6] = n.y * n.z * (1 - c) - n.x * s;
784 	// third line
785 	that.m[8]  = n.z * n.x * (1 - c) - n.y * s;
786 	that.m[9]  = n.z * n.y * (1 - c) + n.x * s;
787 	that.m[10] = n.z * n.z * (1 - c) + c;
788 	
789 	that.mul (this);
790 	this.m = that.m
791 	return this;
792 };

796 /**
797  * Transforms a matrix by modifying the line "which".
798  * 
799  * @param {int} which - which coordinate is changed
800  * @param {int} from - using this one.
801  * @param {int} value - multiplication factor.
802  * 
803  * @return {Matrix} the new matrix.
804  * @throw {String} The parameters are out of wrong value (out of bounds)
805  */
806 Matrix.prototype.shearing = function (which, from, value) {
807 	if (from < 0 || from > 3 || which < 0 || which > 3 || from == which) {
808 		throw "Matrix.shearing: bad type(s) of parameter(s)";
809 	}
810 	// same matrice, except the line "which"
811 	for (var c = 0; c < 4; ++c) {
812 		this.set (from, c, this.get (from, c) + value * this.get (which, from));
813 	}
814 	// mandatory
815 	return this;
816 };
