 47 /** 
 48  * @constructor
 49  */
 50 function Point () {
 51 	Vector.call (this);
 52 	switch (arguments.length) {
 53 		case 0 : 
 54 			return this.defaultConstructor ();
 55 		case 1 :
 56 			if (arguments[0] instanceof Array)
 57 				return this.arrayToPointConstructor (arguments[0]);
 58 			else if (typeof arguments[0] == "number")
 59 				return this.coordinateConstructor (arguments[0], arguments[0]);
 60 			else
 61 				return this.copyConstructor (arguments[0]);
 62 		case 2 : 
 63 			return this.coordinateConstructor (arguments[0], arguments[1]);
 64 		default: 
 65 			throw "Point constructor bad syntax";
 66 	};
 67 	return this;
 68 };

 73 /**
 74  * Default constructor (no argument).
 75  * @return {Point} a Point initialized with 0 in each dimension.
 76  */
 77 Point.prototype.defaultConstructor = function () {
 78 	this.m[0] = 0;
 79 	this.m[1] = 0;
 80 	this.m[2] = 0;
 81 	return this;
 82 };

 85 //==============================================================================
 86 /**
 87  * Copy constructor (on argument, another Point).
 88  * @param {Point} that - the ¨Point to copy.
 89  * @return {Point} a Point initialized with the value from that.
 90  * @throw {String} The parameter is not of type Point.
 91  */
 92 Point.prototype.copyConstructor = function (that) {
 93 	if (!(that instanceof Point)) { 
 94 		throw("Point: bad copy constructor call");
 95 	}
 96 	this.m[0] = that.m[0];
 97 	this.m[1] = that.m[1];
 98 	return this;
 99 };

102 //==============================================================================
103 /**
104  * Transforms an array of two coordinates to a Point
105  * @param {Point} tab - the Point to copy.
106  * @return {Point} a Point initialized with the value from that.
107  * @throw {String} The parameter is not an array with a length of 2.
108  */
109 Point.prototype.arrayToPointConstructor = function (tab) {
110 	if (tab.length != 2) { 
111 		throw "Point : bad array length in array to point constructor call";
112 	}
113 	return this.coordinateConstructor (tab[0], tab[1]);
114 };

118 /**
119  * Constructor using two coordinates (float values).
120  *
121  * @param {float} x - the first dimension value.
122  * @param {float} y - the second dimension value.
123  * 
124  * @return {Point} a Point initialized with x, y and z.
125  * @throw {String} The parameters are not of finites.
126  */
127 Point.prototype.coordinateConstructor = function (x, y) {
128 	if (isFinite (x) && isFinite (y)) {
129 		this.m[0] = x;
130 		this.m[1] = y;
131 		return this;
132 	}
133 	else {
134 		throw "Point: bad (by coordinate) constructor call (" + x + ", " + y 
135 			+ ", " + z + ")";
136 	}
137 };

278 /**
279  * @override
280  * Return the addition of two point.
281  * 
282  * @param a {Point} - A point.
283  * @param b {Point} - A point.
284  * 
285  * @return {Point} a new point corresponding to "a + b"
286  */
287 function addPoint (a, b) {
288 	if ((!(a instanceof Point) && !(a instanceof Array)) 
289 		|| (!(b instanceof Point) && !(b instanceof Array)))
290 	{
291 		throw "addPoint : one of term is not a Point"
292 	}
293 	var pa = new Point (a);
294 	var pb = new Point (b);
295 	return pa.add (pb);
296 };
