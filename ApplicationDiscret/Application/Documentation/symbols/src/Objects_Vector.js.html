106 /**
107  * @constructor
108  * Depending on the number of argument to the constructor, call for
109  * different vector construction.
110  * 	0 : Default constructor : 3 dimensions vector
111  * 	1 : Copy constructor : copy of the vector in argument
112  * 	3 : Creation of a 3 dimensions vector initialized with the arguments
113  * 	defaut : Error
114  *
115  * @return {Vector} the new vector
116  */
117 function Vector () {
118 
119 	/**
120 	 * {VEC_ARRAY_TYPE} The main array which containt the coordinates.
121 	 */
122 	this.m = new VEC_ARRAY_TYPE (3);
123 
124 	/**
125 	 * {Number} An alias to the 1st coordinate.
126 	 */
127 	Object.defineProperty (this, "x", {
128 		get : function () {return this.m[0]},
129 		set : function (value) {this.m[0] = value}
130 	});
131 
132 	/**
133 	 * {Number} An alias to the 2nd coordinate.
134 	 */
135 	Object.defineProperty (this, "y", {
136 		get : function () {return this.m[1]},
137 		set : function (value) {this.m[1] = value}
138 	});
139 
140 	/**
141 	 * {Number} An alias to the 3rd coordinate.
142 	 */
143 	Object.defineProperty (this, "z", {
144 		get : function () {return this.m[2]},
145 		set : function (value) {this.m[2] = value}
146 	});
147 	try {
148 		switch (arguments.length) {
149 			case 0:
150 				return this.defaultConstructor ();
151 			case 1:
152 				if (arguments[0] instanceof Array
153 					|| arguments[0] instanceof Float32Array)
154 				{
155 					return this.arrayToVectorConstructor (arguments[0]);
156 				}
157 				else if (typeof arguments[0] == "number")
158 					return this.coordinateConstructor (arguments[0],
159 						arguments[0], arguments[0])
160 				else
161 					return this.copyConstructor (arguments[0]);
162 			case 3:
163 				return this.coordinateConstructor (arguments[0],
164 					arguments[1], arguments[2]);
165 			default:
166 				throw "Vector constructor bad syntax";
167 		};
168 	}
169 	catch (e) {
170 		console.error (e);
171 	}
172 //	return this;
173 };

177 /**
178  * Default constructor (no argument).
179  *
180  * @return {Vector} A vector initialized with 0 in each coordinate.
181  */
182 Vector.prototype.defaultConstructor = function () {
183 	this.m[0] = 0;
184 	this.m[1] = 0;
185 	this.m[2] = 0;
186 	return this;
187 };

190 //==============================================================================
191 /**
192  * Copy constructor.
193  *
194  * @param {Vector} that - The vector to copy.
195  *
196  * @return {Vector} A vector initialized with the value from 'that'.
197  * @throw {String} The parameter is not of type Vector
198  */
199 Vector.prototype.copyConstructor = function (that) {
200 	if (!(that instanceof Vector)) {
201 		throw "Vector: bad copy constructor call";
202 	}
203 	this.m[0] = that.m[0];
204 	this.m[1] = that.m[1];
205 	this.m[2] = that.m[2];
206 	return this;
207 };

210 //==============================================================================
211 /**
212  * Transforms an array to a Vector.
213  *
214  * @param {Vector} tab - The vector to copy.
215  *
216  * @return {Vector} A vector initialized with the value from 'that'.
217  * @throw {String} The array does not have a length of 3
218  */
219 Vector.prototype.arrayToVectorConstructor = function (tab) {
220 	if (tab.length != 3) {
221 		throw "Vector: bad array length in array to vector constructor call";
222 	}
223 	return this.coordinateConstructor (tab[0], tab[1], tab[2]);
224 };

227 //==============================================================================
228 /**
229  * Constructor using three coordinates (float values).
230  *
231  * @param {float} x - The first dimension value.
232  * @param {float} y - The second dimension value.
233  * @param {float} z - The third dimension value.
234  *
235  * @return {Vector} A vector initialized with x, y and z.
236  */
237 Vector.prototype.coordinateConstructor = function (x, y, z) {
238 	if (isFinite (x) && isFinite (y) && isFinite (z)) {
239 		this.m[0] = x;
240 		this.m[1] = y;
241 		this.m[2] = z;
242 		return this;
243 	}
244 	else {
245 		throw "Vector: bad coordinate constructor call (" + x + "; " + y
246 			+ "; " + z + ")";
247 	}
248 };

258 /**
259  * @return {Number} The euclidean length of the vector.
260  */
261 Vector.prototype.getLength = function () {
262 	return Math.sqrt (
263 		this.m[0] * this.m[0]
264 		+ this.m[1] * this.m[1]
265 		+ this.m[2] * this.m[2]
266 	);
267 };

287 /**
288  * Addition (this += that).
289  *
290  * @param {Vector} that - The vector to add with.
291  *
292  * @return {Vector} The vector result of the addition.
293  * @throws {String} If the 'that' is not a Vector.
294  */
295 Vector.prototype.add = function (that) {
296 	if (! that instanceof Vector) {
297 		throw "Vector.add: given parameter is not a Vector"
298 	}
299 	this.m[0] += that.m[0];
300 	this.m[1] += that.m[1];
301 	this.m[2] += that.m[2];
302 	return this;
303 };

307 /**

309  *
310  * @param {Vector} that - The vector to substract with.

312  * @return {Vector} The vector result of the substraction.
313  * @throws {String} If the 'that' is not a Vector.
314  */
315 Vector.prototype.sub = function (that) {
316 	if (! that instanceof Vector) {
317 		throw "Vector.sub: given parameter is not a Vector"
318 	}

320 	this.m[1] -= that.m[1];
321 	this.m[2] -= that.m[2];
322 	return this;

326 //==============================================================================
327 /**
328  * Cross product (returns = this ^ that).
329  *
330  * @param {Vector} that - The vector to realize the cross product with.
331  *
332  * @return {Vector} The vector result of the cross product.
333  * @throws {String} If the 'that' is not a Vector.
334  */
335 Vector.prototype.cross = function (that) {
336 	if (! that instanceof Vector) {
337 		throw "Vector.cross: given parameter is not a Vector"
338 	}
339 	return new Vector (
340 		this.m[1] * that.m[2] - this.m[2] * that.m[1],
341 		this.m[2] * that.m[0] - this.m[0] * that.m[2],
342 		this.m[0] * that.m[1] - this.m[1] * that.m[0]
343 	);
344 };

347 //==============================================================================
348 /**
349  * Dot product (returns = this . that).
350  *
351  * @param {Vector} that - The vector to realize the dot product with.
352  *
353  * @return {Number} A scalar result of the dot product.
354  * @throws {String} If the 'that' is not a Vector.
355  */
356 Vector.prototype.dot = function (that) {
357 	if (! that instanceof Vector) {
358 		throw "Vector.dot: given parameter is not a Vector"
359 	}
360 	return this.m[0] * that.m[0] + this.m[1] * that.m[1]
361 		+ this.m[2] * that.m[2];
362 };

388 /**
389  * Normalization. This becomes unit vector.
390  *
391  * @return {Vector} The vector normalized.
392  */
393 Vector.prototype.normalize = function () {
394 	var norm = Math.sqrt (this.m[0] * this.m[0] + this.m[1] * this.m[1]
395 		+ this.m[2] * this.m[2]);
396 	if (norm !== 0) {
397 		this.m[0] /= norm;
398 		this.m[1] /= norm;
399 		this.m[2] /= norm;
400 	}
401 	return this;
402 };

418 /**
419  * Rotate a vector given an angle (in radian) around X axis.
420  *
421  * @param {float} rad - The angle of rotation (in radian).
422  *
423  * @return {Vector} The vector after the rotation.
424  */
425 Vector.prototype.rotateX = function (rad) {
426 	return new Vector (
427 		this.m[0],
428 		this.m[1] * Math.cos (rad) - this.m[2] * Math.sin (rad),
429 		this.m[1] * Math.sin (rad) + this.m[2] * Math.cos (rad)
430 	);
431 };

520 /**
521  * Returns the addition of two vector.
522  *
523  * @param {(Vector | Number[3] | Number)} a - A vector or coordinates to
524  * construct a vector.
525  * @param {(Vector | Number[3] | Number)} b - A vector or coordinates to
526  * construct a vector.
527  * @see {@link copyConstructor, arrayToVectorConstructor, coordinateConstructor}
528  *
529  * @return {Vector} A new vector corresponding to "a + b"
530  * @throw {String} The parameters are not of expected types.
531  */
532 function addVector (a, b) {
533 	if (! checkType (arguments, [Vector, Array, "number"],
534 		[Vector, Array, "number"]))
535 	{
536 		throw "addVector: one of parameter is not a Vector"
537 	}
538 	var va = new Vector (a);
539 	var vb = new Vector (b);
540 	return va.add (vb);
541 };
