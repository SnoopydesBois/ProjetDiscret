 50 /**
 51  * {String} Vertex shader for the application.
 52  */
 53 var vertSrc =
 54   "attribute vec3 aPosition;\n"
 55 + "attribute vec4 aColor;\n"
 56 + "\n"
 57 + "uniform mat4 uOrthographicProjectionMatrix;\n"
 58 + "uniform mat4 uPerspectiveProjectionMatrix;\n"
 59 + "uniform mat4 uModelViewMatrix;\n"
 60 + "uniform int uViewMode;\n"
 61 + "\n"
 62 + "varying vec4 vColor;\n"
 63 + "varying vec3 vPos;\n"
 64 + "\n"
 65 + "void main (void) {\n"
 66 + "  if (uViewMode == 1) {\n" // Perspective
 67 + "    gl_Position = uPerspectiveProjectionMatrix * uModelViewMatrix\n"
 68 + "      * vec4 (aPosition, 1.0);\n"
 69 + "  }\n"
 70 + "  else {\n" // Orthographic
 71 + "    gl_Position = uOrthographicProjectionMatrix * uModelViewMatrix\n"
 72 + "      * vec4 (aPosition, 1.0);\n"
 73 + "  }\n"
 74 + "	 vPos = aPosition;\n"
 75 + "	 vColor = aColor;\n"
 76 + "}";
