 67 /**
 68  * @constructor 
 69  * A ExplicitAlgo1Worker contains a number of workers that will
 70  * compute the surface with the Explicit algo with no optimization.
 71  * 
 72  * @param {Equation} explicitCurve - The equation for the meridian.
 73  * @param {Equation} implicitCurve - The equation for the revolution curve.
 74  * @param {Vector} dimension - The space dimensions.
 75  * @param {Surface} surface - The surface to draw.
 76  */
 77 function ExplicitAlgo1Worker (explicitCurve, implicitCurve, dimension, surface){
 78 	AlgoWorker.call (this, explicitCurve, implicitCurve, dimension, surface);
 79 	for (var i = 0; i < 8; ++i) {
 80 		this.worker[i] = new Worker ("Js/Threads/EA1Worker.js");
 81 		this.activeWorkers++;
 82 		var that = this
 83 		this.worker[i].onmessage = function(e) {
 84 			if (e.data[0] === "Abort") {
 85 				that.finished = "error"
 86 				appli.alertMessage ("Aborted");
 87 			}
 88 			else if (e.data[0] === "Terminate") {
 89 				that.worker[e.data[1]].terminate();
 90 				that.worker[e.data[1]] = undefined;
 91 				--that.activeWorkers;
 92 				if (that.activeWorkers == 0 && that.finished != "error") {
 93 					that.finished = true;
 94 				}
 95 			}
 96 			else {
 97 				that.readBuffer(e.data[0], e.data[1]);
 98 			}
 99 		};
100 		this.worker[i].postMessage([i, this.meridianCurve,
101 			this.revolutionCurve, this.dim,
102 			i * Math.floor ((this.dim[2] + 7) / 8),
103 			(i + 1) * Math.floor ((this.dim[2] + 7) / 8)
104 		]);
105 	}
106 }
