 86 /**
 87  * @constructor
 88  *
 89  * @param {Number} imageMin - The lower bound of the output range.
 90  * @param {Number} imageMax - The upper bound of the output range.
 91  * @param {Number} antecedantMin - The lower bound of the input range.
 92  * @param {Number} antecedantMax - The upper bound of the input range.
 93  * @param {DrawModeEnum} mode - The drawing method used.
 94  */
 95 function Controller2D (imageMin, imageMax, antecedantMin, antecedantMax, mode) {
 96 	if (arguments.length == 0) {
 97 		Controller.call (this);
 98 	}
 99 	else {
100 		if (! checkType (arguments, "number", "number", "number", "number",
101 			"number"))
102 		{
103 			console.error ("Controller2D.constructor: bad type(s) of "
104 				+ "parameter(s)");
105 			showType ();
106 		}
107 
108 		Controller.call (this);
109 
110 		/**
111 		 * {ModelCurve} The model that creates curve.
112 		 */
113 		this.modelCurve = new ModelCurve (
114 			new Range (imageMin, imageMax),
115 			new Range (antecedantMin, antecedantMax)
116 		);
117 
118 		/**
119 		 * {DrawModeEnum} The type of drawing method used.
120 		 */
121 		this.mode = mode;
122 		
123 		/**
124 		 * {ModelParameter} The model that controls the curve parameters
125 		 */
126 		this.modelParameter = new ModelParameter();
127 	}
128 };

138 /**
139  * Add a curve to the model.
140  * @see {@link ModelCurve.addCurve}
141  *
142  * @param {String} name - The name of the curve.
143  * @param {Function} constructor - The constructor of the curve. This class
144  * constructor must inherit from Curve.
145  *
146  * @return {void}
147  */
148 Controller2D.prototype.addCurve = function (name, constructor) {
149 	/// parameter verification
150 	if (! checkType (arguments, "string", Function)) {
151 		showType (name, constructor);
152 		throw "Controller2D.addCurve: bad type(s) of parameter(s)";
153 	}
154 
155 	/// add the curve
156 	this.modelCurve.addCurve (name, constructor);
157 };

168 };

171 //==============================================================================
172 /**
173  * Set the active curve
174  *
175  * @param {(Curve | String)} name - If of type curve, it is the new curve to set as the active curve
176  * if of type String, it is the name of a predefined curve to set as active
177  * @param {(EquationTypeEnum)} type - If name is of type Curve, the model needs to know the
178  * type of this new curve
179  *
180  * @return {void}
181  */
182 Controller2D.prototype.setActive = function (name, type) {
183 	if(!(name instanceof Curve) && !(typeof name == "string")){
184 		throw "Controller2D.setActive.ErrorParameterType";
185 	}
186 
187 	/// set the curve
188 	var curve;
189 	if(type == undefined){
190 		curve =  this.modelCurve.setActive (name);
191 	}
192 	else{
193 		curve = this.modelCurve.setActive (name, type);
194 	}
195 	this.modelParameter.setCurve(curve);
196 };
