 61 /**
 62  * @classdesc This class inherits from Controller2D and is used to communicate with a modelCurve or a ModelDraw
 63  */
 64 Controller2DMeridian.prototype = new Controller2D;
 65 Controller2DMeridian.prototype.constructor = Controller2DMeridian;
 66

 68 
 69 //##############################################################################
 70 //	Constructor
 71 //##############################################################################
 72 
 73 
 74 
 75 /**
 76  * @constructor
 77  *
 78  * @param {Number} imageMin - The lower bound of the output range.
 79  * @param {Number} imageMax - The upper bound of the output range.
 80  * @param {Number} antecedantMin - The lower bound of the input range.
 81  * @param {Number} antecedantMax - The upper bound of the input range.
 82  * @param {DrawModeEnum} mode - The drawing method used.
 83  */
 84 function Controller2DMeridian (imageMin, imageMax, antecedantMin, antecedantMax,
 85 	mode)
 86 {
 87 	// Call to mother constructor
 88 	Controller2D.call (this, imageMin, imageMax, antecedantMin, antecedantMax,
 89 		DrawModeEnum.HAND_FREE);
 90 
 91 	/**
 92 	 * {ModelDraw} The model used to handle the drawn curve
 93 	 */
 94 	this.modelDraw = new ModelDraw ();
 95 
 96 	/**
 97 	 * {HTMLSelectElement} The current curve mode.
 98 	 */
 99 	this.mode = document.getElementById ("meridianType");
100 };
101

106 //##############################################################################
107 
108 /**
109  * Set an active curve depending of the active mode
110  * @param {(Curve | String)} name - If of type curve, it is the new curve for the model
111  * if of type String, it is the name of a predefined curve to set as active
112  * @param {(EquationTypeEnum)} type - If name is of type Curve, the model needs to know the
113  * type of this new curve
114  * @return {void}
115  */
116 Controller2DMeridian.prototype.setActive = function (name, type) {
117 	var mode = document.forms["meridianType"]["meridianTypeValue"].value
118 	var curve;
119 	// List of curves mode
120 	if (mode === "meridianPrimitive") {
121 		curve = this.modelCurve.setActive (name, type);
122 		this.modelParameter.setCurve(curve);
123 	}
124 	// Draw mode
125 	else if (mode === "meridianFreeHand") {
126 		curve = this.newCurve ();
127 	}
128 	// Formula mode
129 	else if (mode === "meridianFormula") {
130 		curve = this.modelCurve.setActive(name, EquationTypeEnum.EXPLICIT);
131 		this.modelParameter.setCurve(curve);
132 	}
133 	else {
134 		console.error ("Controller2DMeridian.getActiveCurve: unknown drawing "
135 			+ "mode: " + mode);
136 	}
137 };
138 
139 //==============================================================================
140 /**
141  * @return {Curve} The current curve.
142  */
143 Controller2DMeridian.prototype.getActiveCurve = function () {
144 	var mode = document.forms["meridianType"]["meridianTypeValue"].value;
145 	if (mode === "meridianPrimitive") {
146 		return this.modelCurve.getActiveCurve ();
147 	}
148 	else if (mode === "meridianFreeHand") {
149 		return this.modelDraw.getActiveCurve ();
150 	}
151 	else if (mode === "meridianFormula") {
152 		return this.modelCurve.getActiveCurve ();
153 	}
154 	else {
155 		console.error ("Controller2DMeridian.getActiveCurve: unknown drawing "
156 			+ "mode: " + mode);
157 		return null;
158 	}
159 };

163 /**
164  * Ask the modelDraw to create a new Curve
165  */
166 Controller2DMeridian.prototype.newCurve = function () {
167 	this.modelDraw.newCurve ();
168 };

177 Controller2DMeridian.prototype.closeCurve = function () {
