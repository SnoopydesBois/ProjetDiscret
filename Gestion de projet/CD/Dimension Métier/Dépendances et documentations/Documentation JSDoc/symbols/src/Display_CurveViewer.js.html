105 /**
106  * @constructor
107  * Constructs a CurveViewer with a controller, a div and a canvas for drawing.
108  * Calls the super constructor with the canvas and the string "2d".
109  * @see {@link GenericViewer}
110  * 
111  * @param {HTMLCanvasElement} canvas - The associated canvas.
112  * @param {HTMLDivElement} div - The associated div where the SVG curve is drawn.
113  * @param {Controller2D} curveController - Controller of this view.
114  */
115 function CurveViewer (canvas, div, curveController) {
116 	GenericViewer.call (this, canvas, "2d");
117 
118 	/**
119 	 * {Controller2D} Controller of this view. Must be a Controller2DMeridian if
120 	 * you want to be able to draw a curve.
121 	 */
122 	this.controller = curveController;
123 
124 	/**
125 	 * {Point} Last point added to the curve. If -1, there are not last
126 	 * points.
127 	 */
128 	this.lastPoint = new Point (-1, -1);
129 
130 	/**
131 	 * {HTMLInputElement} Input which contains the Y dimension of the 3D space.
132 	 * Used to compute the x range of the drawn curve.
133 	 */
134 	this.xMaxInput = document.getElementById ("dimy");
135 
136 	/**
137 	 * {HTMLInputElement} Input which contains the Z dimension of the 3D space.
138 	 * Used to compute the y range of the drawn curve.
139 	 */
140 	this.yMaxInput = document.getElementById ("dimz");
141 
142 	/**
143 	 * {float} Minimal distance between two added points.
144 	 */
145 	this.MIN_DIST_BETWEEN_POINT = 0.5;
146 
147 	/**
148 	 * {HTMLFormElement} Use 'value' attribute to identify the drawing mode (
149 	 * i.e. primitive, free hand or formula).
150 	 * @see {@link onMouseDown, onMouseMove}
151 	 */
152 	this.formModeSelected = document.forms["meridianType"];
153 	// FIXME just adapt to the meridian
154 	
155 	/**
156 	 * {HTMLDivElement} The div where the SVG (i.e. explicit and implicit curve)
157 	 * is drawn.
158 	 */
159 	this.displayDiv = div;
160 	
161 	
162 	// initialisation
163 	this.initCanvasEvent ();
164 	this.resizeCanvas ();
165 };

200 /**
201  * Draws the curve on the target (canvas or div).
202  *
203  * @return {void}
204  */
205 CurveViewer.prototype.draw = function () {
206 	var curve = this.controller.getActiveCurve ();
207 	var xRange = this.controller.getXRange ();
208 	if (curve instanceof DrawnCurve) {
209 		/// set the canvas size
210 		this.resizeCanvas ();
211 		/// draw the curve
212 		this.drawFreeHand (curve);
213 	}
214 	else if (curve.getEquation ().toString () == "undefined") {
215 		this.drawGrid ();
216 	}
217 	else if (curve instanceof ImplicitCurve) {
218 		this.drawImplicit (curve);
219 	}
220 	else if (curve instanceof ExplicitCurve) {
221 		this.drawExplicit (curve);
222 	}
223 	else {
224 		this.drawGrid ();
225 		console.error ("Bad type of curve, find: " + type (curve));
226 	}
227 };

337 /**
338  * Draws a free hand curve on the canvas.
339  *
340  * @param {DrawnCurve} curve - The curve to draw.
341  *
342  * @return {void}
343  */
344 CurveViewer.prototype.drawFreeHand = function (curve) {
345 	/// Let's render
346 	var x = curve.getXList (),
347 		y = curve.getYList ();
348 	var len = x.length, point;
349 
350 	this.glContext.clearRect (0, 0, this.glContext.canvas.width,
351 		this.glContext.canvas.height);
352 	if (len > 0) {
353 		this.lastPoint = new Point (x[0], y[0]);
354 		for (var i = 1; i < len ; ++i) {
355 			point = new Point (x[i], y[i]);
356 			this.drawSegment (this.lastPoint, point);
357 			delete this.lastPoint;
358 			this.lastPoint = point;
359 		} // end for each next point
360 	} // end if there are point
361 };

365 /**
366  * Draws a segment between two points. /!\ Arguments are the curve's points not the pixel's
367  * points.
368  *
369  * @param {Point} pointA - The first point.
370  * @param {Point} pointB - The second point.
371  *
372  * @return {void}
373  * @throws {String} One of the parameters is not of type Point.
374  */
375 CurveViewer.prototype.drawSegment = function (pointA, pointB) {
376 	/// parameter verification
377 	if (! checkType (arguments, Point, Point)) {
378 		throw "CurveViewer.drawSegment: one of parameter is not a Point";
379 	}
380 
381 	var p1 = this.pointToPixel (pointA.x, pointA.y),
382 		p2 = this.pointToPixel (pointB.x, pointB.y);
383 
384 	/// draw it
385 	var ctx = this.glContext;
386 	ctx.strokeStyle = "black";
387 	ctx.beginPath ();
388 	ctx.moveTo (p1.x, p1.y);
389 	ctx.lineTo (p2.x, p2.y);
390 	ctx.stroke ();
391 };

409 /**
410  * Resizes the associated canvas.
411  *
412  * @return {void}
413  */
414 CurveViewer.prototype.resizeCanvas = function () {
415 	var $ref = $("#meridianCanvas2"); // FIXME just adapt to the meridian
416 	var max = Math.max (this.xMaxInput.value / 2, this.yMaxInput.value);
417 	var canvas = this.glContext.canvas;
418 
419 	canvas.width = ($ref.width () + 2) * this.xMaxInput.value / 2 / max;
420 	canvas.style.width = canvas.width + "px";
421 	canvas.style.right = (($ref.width () + 2) - canvas.width) / 2 + "px";
422 
423 	canvas.height = ($ref.height () + 2) * this.yMaxInput.value / max;
424 	canvas.style.height = canvas.height + "px";
425 	canvas.style.top = (($ref.height () + 2) - canvas.height) / 2 + "px";
426 };

535 /**
536  * Transforms a pixel coordinates on the canvas into point of the curve. /!\ The
537  * origin of the point is the bottom left corner but pixel's origin is the top left
538  * corner. Top and bottom are inverted in this function.
539  *
540  * @param {float} x - Pixel X coordinate.
541  * @param {float} y - Pixel Y coordinate.
542  *
543  * @return {Point} The computed point (truncate to two decimal number).
544  * @throws {String} One of the parameter is not of type Number.
545  */
546 CurveViewer.prototype.pixelToPoint = function (x, y) {
547 	/// parameter verification
548 	if (! checkType (arguments, "number", "number")) {
549 		throw "CurveViewer.pixelToPoint: bad type(s) of parameter(s)";
550 	}
551 
552 	/// compute
553 	var point = new Point (
554 		x * (this.xMaxInput.value / 2) / (this.glContext.canvas.width + 1),
555 		(y + 1) * this.yMaxInput.value / (this.glContext.canvas.height + 1)
556 	);
557 	point.y = this.yMaxInput.value - point.y;
558 	return point;
559 };

562 //==============================================================================
563 /**
564  * Transforms a point of the curve into pixel coordinates on the canvas. /!\ The
565  * origin of the point is the bottom left corner but pixel's origin is the top left
566  * corner. Top and bottom was inverted in this function.
567  *
568  * @param {float} x - Point X coordinate.
569  * @param {float} y - Point Y coordinate.
570  *
571  * @return {Point} The computed pixel.
572  * @throws {String} One of the parameter is not of type Number.
573  */
574 CurveViewer.prototype.pointToPixel = function (x, y) {
575 	/// parameter verification
576 	if (! checkType (arguments, "number", "number")) {
577 		throw "CurveViewer.pointToPixel: bad type(s) of parameter(s)";
578 	}
579 
580 	/// compute
581 	var pixel = new Point (
582 		x * this.glContext.canvas.width / (this.xMaxInput.value / 2),
583 		y * this.glContext.canvas.height / this.yMaxInput.value
584 	);
585 	pixel.y = this.glContext.canvas.height - 1 - Math.floor (pixel.y);
586 	return pixel;
587 };

590 //==============================================================================
591 /**
592  * Adds a point to the current curve and draws it on the canvas. If the point is
593  * too close to the last, it is not added.
594  *
595  * @param {Point} point - A point.
596  *
597  * @return {boolean} True if the point was added, false otherwise.
598  * @throw {String} the parameter is not of type Point
599  */
600 CurveViewer.prototype.addPoint = function (point) {
601 	/// parameter verification
602 	if (! point instanceof Point) {
603 		throw "CurveViewer.addPoint: parameter is not a Point";
604 	}
605 
606 	/// add
607 	if (Math.hypot (point.x - this.lastPoint.x, point.y - this.lastPoint.y)
608 		 > this.MIN_DIST_BETWEEN_POINT)
609 	{
610 		this.controller.getActiveCurve ().addPoint (point.x, point.y);
611 		if (this.lastPoint.x != -1)
612 			this.drawSegment (this.lastPoint, point);
613 		this.lastPoint = point;
614 		return true;
615 	}
616 	else
617 		return false;
618 };

622 /**
623  * @static
624  * Computes the Y range of a curve with its X range.
625  *
626  * @param {int} width - Width in pixels of the HTML element where the curve is
627  * displayed.
628  * @param {int} height - Height in pixels of the HTML element where the curve is
629  * displayed.
630  * @param {Range} xRange - X range of the curve.
631  *
632  * @return {int[2]} Y range of the curve.
633  */
634 CurveViewer.computeYScale = function (width, height, xRange) {
635 	var xDiff = xRange.length();
636 	var yDiff = height * xDiff / width;
637 	return [-yDiff / 2, yDiff / 2];
638 };

647 CurveViewer.prototype.closeCurve = function () {

651 		this.lastPoint = addedPoint;
652 	}
