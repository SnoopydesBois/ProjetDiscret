103 /**
104  * @constructor
105  * Creates a 3D container.
106  */
107 function Scene () {
108 
109 	GenericContainer.call (this);
110 
111 	/**
112 	 * {Vector} Initial camera position.
113 	 */
114 	this.defaultCameraPosition = new Vector (3.5, 3.0, 2.5);
115 
116 	/**
117 	 * {Vector} Initial camera lookAt point.
118 	 */
119 	this.defaultLookAtPoint = new Vector (0, 0, 0);
120 
121 	/**
122 	 * {Camera} The camera used in the scene.
123 	 */
124 	this.camera = new Camera (
125 		new Vector (this.defaultCameraPosition),
126 		new Vector (this.defaultLookAtPoint),
127 		new Vector (0, 0, 1),
128 		800,
129 		600,
130 		45.0,
131 		0.0001,
132 		30.0
133 	);
134 
135 	/**
136 	 * {int} The X mouse coordinate.
137 	 */
138 	this.mouseX = 512;
139 
140 	/**
141 	 * {int} The Y mouse coordinate.
142 	 */
143 	this.mouseY = 384;
144 };

189 /**
190  * @override
191  *
192  * Adds an object (only if it is a subclass of GenericStructure).
193  *
194  * @param {!GenericStructure} anObject - Object to add to the scene.
195  *
196  * @return {void}
197  * @throws {String} The provided parameter is not of type GenericStructure or
198  * a GenericStructure subclass.
199  */
200 Scene.prototype.addObject = function (anObject) {
201 	if (anObject instanceof GenericStructure) {
202 		this.objectList.push (anObject);
203 	}
204 	else
205 		throw "Scene.addObject: parameter is not a GenericStructure";
206 };

209 //==============================================================================
210 /**
211  * Gets an object given its name.
212  *
213  * @param {String} aName - Name of the object to return.
214  *
215  * @return {GenericStructure} The object corresponding to the name in parameter
216  * if it exists, null otherwise.
217  */
218 Scene.prototype.getObjectByName = function (aName) {
219 	var length = this.objectList.length;
220 
221 	for (var i = 0; i < length; ++i) {
222 		if (this.objectList[i].getName() === aName)
223 			return this.objectList[i];
224 	}
225 	return null;
226 };
227 
228 
229 //==============================================================================
230 /**
231  * Removes an object by name.
232  *
233  * @param {!String} aName - Name of the object to remove.
234  *
235  * @return {void}
236  */
237 Scene.prototype.removeObjectByName = function (aName) {
238 	for (var i = 0; i < this.objectList.length; ++i) {
239 		if (this.objectList[i].getName() === aName) {
240 			this.objectList.splice (i, 1); // Remove from the list
241 			return;
242 		}
243 	}
244 };

254 	var len = this.getNbObject ();

268 /**
269  * Reloads the scene. Reloads shader for each displayable object.
270  *
271  * @return {void}
272  */
273 Scene.prototype.reload = function () {
274 	var obj, length = this.getNbObject ();
275 
276 	for (var i = 0; i < length; ++i) {
277 		obj = this.objectList[i];
278 		if (obj.displayMe ())
279 			obj.getShader ().reload ();
280 	}
281 };

285 /**
286  * Prepares the scene before rendering. Call the 'prepare' method of every object.
287  *
288  * @param {WebGLRenderingContext} glContext - The gl context.
289  * @param {ConnexityEnum} connexity - Which connexity is displayed.
290  * @param {float} voxelRadius - The voxel radius.
291  *
292  * @return {void}
293  */
294 Scene.prototype.prepare = function (glContext, connexity, voxelRadius) {
295 	var length = this.getNbObject ();
296 
297 	for (var i = 0; i < length; ++i) {
298 		if (! this.objectList[i].isPrepared ()) {
299 			if (this.objectList[i] instanceof SurfaceRenderer) { // FIXME sale !
300 				var o = this.objectList[i].getModelController ().getSurface ();
301 				o.printOnly (
302 					new Range (
303 						parseInt ($("#slider-rangeX").slider ('values', 0)),
304 						parseInt ($("#slider-rangeX").slider ('values', 1))
305 							- 1
306 					),
307 					new Range (
308 						parseInt ($("#slider-rangeY").slider ('values', 0)),
309 						parseInt ($("#slider-rangeY").slider ('values', 1))
310 							- 1
311 					),
312 					new Range (
313 						parseInt ($("#slider-rangeZ").slider ('values', 0)),
314 						parseInt ($("#slider-rangeZ").slider ('values', 1))
315 							- 1
316 					)
317 				);
318 			} // fin du truc sale
319 			this.objectList[i].prepare (glContext, connexity, voxelRadius);
320 			this.objectList[i].getShader ().activate ();
321 		} // end if not prepared
322 	} // end for all object
323 };

327 /**
328  * Draws a scene.
329  *
330  * @param {WebGLRenderingContext} glContext - The gl context.
331  * @param {boolean} backBuffer - Indicates if we have to draw the scene
332  * normally or if we need to draw for picking (false -> normally; 
333  * true -> picking).
334  * @param {float} voxelRadius - Voxels' radius.
335  * @param {WebGLRenderbuffer} renderBuffer - The render buffer.
336  *
337  * @return {void}
338  */
339 Scene.prototype.draw = function (
340 	glContext,
341 	backBuffer,
342 	voxelRadius,
343 	renderBuffer)
344 {
345 	var size = Math.min (this.height, this.width) * 2;
346 	glContext.clear (glContext.COLOR_BUFFER_BIT | glContext.DEPTH_BUFFER_BIT);
347 	glContext.bindRenderbuffer (glContext.RENDERBUFFER, renderBuffer);
348 	var length = this.getNbObject ();
349 	for (var i = 0; i < length; ++i) {
350 		// Get Object Properties
351 		var obj = this.objectList[i];
352 		if (!obj.displayMe ())
353 			continue;
354 
355 		this.prepareShaderMatrix (glContext, obj);
356 
357 		// Render Object
358 		if (backBuffer) {
359 			try {
360 				obj.drawBackBuffer (glContext);
361 			}
362 			catch (e) {
363 //				console.warn (obj.getName() + ": object is not pickable !");
364 			}
365 		}
366 		else
367 			obj.draw (glContext, voxelRadius);
368 	} // end for each displayable object
369 };

373 /**
374  * Computes matrices (model, view, projection) of an object and send it to the
375  * shader.
376  *
377  * @param {WebGLRenderingContext} glContext - The gl context.
378  * @param {GenericStructure} obj - An object.
379  *
380  * @return {void}
381  * @throw {String} The parameters are not of correct types
382  */
383 Scene.prototype.prepareShaderMatrix = function (glContext, obj) {
384 	/// Parameter verification
385 	if (! (glContext instanceof WebGLRenderingContext
386 		&& obj instanceof GenericStructure))
387 	{
388 		showType (glContext, obj);
389 		throw "Scene.prepareShaderMatrix: bad type(s) of parameter(s)"
390 	}
391 
392 	var cam = this.camera;
393 	var mvMat = cam.getViewMatrix ();
394 	var ppMat = cam.getPerspectiveProjectionMatrix ();
395 	var opMat = cam.getOrthographicProjectionMatrix ();
396 	var objMat = new Matrix (obj.getMatrix (cam));
397 
398 	// Get Location of uniform variables
399 	var shad = obj.getShader ();
400 	shad.activate (glContext);
401 	var locMvMat = shad.getUniformLocation ("uModelViewMatrix");
402 	var locPpMat = shad.getUniformLocation ("uPerspectiveProjectionMatrix");
403 	var locOpMat = shad.getUniformLocation ("uOrthographicProjectionMatrix");
404 	var locDim = shad.getUniformLocation ("uDimension");
405 
406 	// Compute real ModelView matrix
407 	var mv = new Matrix (mvMat).mul (objMat);
408 
409 	// Set Uniform Matrices
410 	if (locMvMat != null)
411 		glContext.uniformMatrix4fv (locMvMat, false, mv.getGLVector ());
412 
413 	if (locPpMat != null)
414 		glContext.uniformMatrix4fv (locPpMat, false, ppMat.getGLVector ());
415 
416 	if (locOpMat != null)
417 		glContext.uniformMatrix4fv (locOpMat, false, opMat.getGLVector ());
418 
419 	if (locDim != null)
420 		glContext.uniform3fv (locDim, obj.getDimension ().getGLVector ());
421 };

431 /**
432  * Sets the position and the lookAt point of the camera. Recomputes the matrix
433  * camera.
434  *
435  * @param {(Vector | Number[3])} position - The new position of the camera.
436  * @param {(Vector | Number[3])} [lookAt] - The new lookAt point of the camera.
437  *
438  * @return {void}
439  */
440 Scene.prototype.setCameraAt = function (position, lookAt) {
441 	this.camera.eyePos = new Vector (position);
442 	lookAt && (this.camera.centerPos = new Vector (lookAt));
443 	this.camera.computeMatrices ();
444 };

448 /**
449  * Replaces the camera at its initial position.
450  * @see {@link centerCamera, setCameraAt}
451  *
452  * @return {void}
453  */
454 Scene.prototype.resetCamera = function () {
455 	this.setCameraAt (this.defaultCameraPosition, this.defaultLookAtPoint);
456 };

460 /**
461  * Project the camera lookAt point on the Z axis.
462  * @see {@link resetCamera, setCameraAt}
463  *
464  * @return {void}
465  */
466 Scene.prototype.centerCamera = function () {
467 	var lookAt = new Vector (
468 		this.defaultLookAtPoint.x,
469 		this.defaultLookAtPoint.y,
470 		this.getCamera ().getLookAtPosition ().z
471 	);
472 	this.setCameraAt (this.getCamera ().getPosition (), lookAt);
473 };
