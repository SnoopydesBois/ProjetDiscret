106 /**
107  * @constructor
108  * @param {HTMLCanvasElement} canvas - The associated canvas.
109  */
110 function SurfaceViewer (canvas) {
111 	GenericViewer.call (this, canvas, "3d");
112 
113 	/**
114 	 * {Scene} The scene to display the surface.
115 	 */
116 	this.container = new Scene ();
117 	this.container.addObject (new BoundingBox (
118 		new Vector (1, 1, 1),
119 		this.glContext
120 	));
121 	this.container.addObject (new Repere (this.glContext));
122 
123 	/**
124 	 * {int[2]} The position of the mouse on the canvas when the user presses a
125 	 * button (used for the camera mouvement). -1 is the default value (used to
126 	 * resolve conflicts).
127 	 * @see {@link onMouseDown, onMouseMove}
128 	 */
129 	this.mousePosOnPress = [-1, -1];
130 
131 	/**
132 	 * {Vector} Camera position in space when the user presses left mouse button.
133 	 * Used by 'onMouseMove', 'rotateCamera' and 'moveCamera' methods to compute
134 	 * the new camera position.
135 	 * @see {@link onMouseMove, rotateCamera}
136 	 */
137 	this.camPosWhenClick = new Vector (3, 3, 3);
138 
139 	/**
140 	 * {Vector} Centers the camera position in space when the user presses the middle mouse
141 	 * button. Used by 'onMouseMove', 'rotateCamera' and 'moveCamera' methods to
142 	 * compute the new camera position.
143 	 * @see {@link onMouseMove, rotateCamera, moveCamera}
144 	 */
145 	this.camCenterWhenClick = new Vector (0, 0, 0);
146 
147 	/**
148 	 * {HTMLInputElement} The connexity selected by the user.
149 	 */
150 	this.connexityInput = document.getElementById ("connexityChoice");
151 
152 	/**
153 	 * {HTMLInputElement} The voxel radius selected by the user.
154 	 */
155 	this.voxelRadiusInput = document.getElementById ("voxelRadius");
156 
157 	/**
158 	 * {WebGLFrameBuffer} The frame buffer which contains all render buffers.
159 	 */
160 	this.frameBuffer = this.glContext.createFramebuffer ();
161 
162 	/**
163 	 * {WebGLRenderBuffer} The main render buffer. Used to draw the main image.
164 	 */
165 	this.screenBuffer = this.glContext.createRenderbuffer ();
166 
167 	/**
168 	 * {WebGLRenderBuffer} Render buffer for picking. Used to store picking
169 	 * color of each facet.
170 	 */
171 	this.backBuffer = this.glContext.createRenderbuffer ();
172 
173 
174 	/// Initialisation
175 	var gl = this.glContext;
176 
177 	this.initCanvasEvent ();
178 	gl.enable (gl.CULL_FACE);
179 	gl.enable (gl.DEPTH_TEST);
180 	gl.frontFace (gl.CW);
181 
182 	gl.bindFramebuffer (gl.FRAMEBUFFER, this.frameBuffer);
183 
184 	// screen buffer
185 	gl.bindRenderbuffer (gl.RENDERBUFFER, this.screenBuffer);
186 	gl.renderbufferStorage (gl.RENDERBUFFER, gl.RGBA4,
187 		gl.drawingBufferWidth, gl.drawingBufferHeight);
188 	gl.framebufferRenderbuffer (gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0,
189 		gl.RENDERBUFFER, this.screenBuffer);
190 
191 	// back buffer
192 	gl.bindRenderbuffer (gl.RENDERBUFFER, this.backBuffer);
193 	gl.renderbufferStorage (gl.RENDERBUFFER, gl.DEPTH_COMPONENT16,
194 		gl.drawingBufferWidth, gl.drawingBufferHeight);
195 	gl.framebufferRenderbuffer (gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT,
196 		gl.RENDERBUFFER, this.backBuffer);
197 
198 	gl.bindRenderbuffer (gl.RENDERBUFFER, null);
199 	gl.bindFramebuffer (gl.FRAMEBUFFER, null);
200 };

298 /**
299  * Prepares the scene if there are some objects.
300  *
301  * @return {void}
302  */
303 SurfaceViewer.prototype.prepare = function () {
304 	if (this.container.getNbObject () != 0) {
305 		this.container.prepare (
306 			this.glContext,
307 			parseInt (this.connexityInput.value),
308 			parseFloat (this.voxelRadiusInput.value) / 2.0
309 		);
310 	}
311 	else
312 		console.log ("No object to prepare");
313 };

318  * Draws the scene if there are some objects.
319  *
320  * @param {boolean} [backBuffer] - Indicates if we have to draw the scene
321  * normally or if we need to draw for picking.
322  *
323  * @return {void}
324  */
325 SurfaceViewer.prototype.draw = function (backBuffer) {
326 	if (this.container.getNbObject () != 0) {
327 		this.container.draw (
328 			this.glContext,
329 			backBuffer,
330 			parseFloat (this.voxelRadiusInput.value) / 2,
331 			this.screenBuffer
332 		);
333 	}
334 	else
335 		console.log ("No object to draw");
336 };

372 //==============================================================================
373 /**
374  * @override
375  * Left and middle buttons : saves the camera coordinates and the mouse
376  * coordinates.
377  * Right button : highlights X, Y and Z slices where the user clicked. If there is
378  * no facet, highlight is desactivated.
379  * @see {@link camPosWhenClick, mousePosOnPress}
380  *
381  * @param {MouseEvent} event - The event.
382  *
383  * @return {void}
384  */
385 SurfaceViewer.prototype.onMouseDown = function (event) {
386 	if (event.buttons & 5) { // left or middle click
387 		this.camPosWhenClick = this.container.getCamera ().getPosition ();
388 		this.mousePosOnPress[0] = event.layerX;
389 		this.mousePosOnPress[1] = event.layerY;
390 		this.camCenterWhenClick =
391 			this.container.getCamera ().getLookAtPosition ();
392 	}
393 	if (event.buttons === 2) {
394 		// draw the scene for picking
395 		this.container.draw (this.glContext, true, this.backBuffer);
396 		var color = new Uint8Array (4);
397 		this.glContext.readPixels (
398 			event.layerX, this.glContext.drawingBufferHeight - event.layerY,
399 			1, 1,
400 			this.glContext.RGBA, this.glContext.UNSIGNED_BYTE, color
401 		);
402 		var surface = this.getSurfaceRenderer ();
403 		var dim = surface.getDimension ();
404 		var facet = colorToPos (color, dim);
405 		var dir = facet.getDirection ();
406 		var noHighlight;
407 		if (0 <= dir && dir < DirectionEnum.size) {
408 			var voxelPos = facet.getCube ();
409 			surface.setHighlightX (voxelPos.x);
410 			surface.setHighlightY (voxelPos.y);
411 			surface.setHighlightZ (voxelPos.z);
412 		}
413 		else {
414 			noHighlight = (surface.highlightX == -1);
415 			if (! noHighlight) {
416 				surface.setHighlightX (-1);
417 				surface.setHighlightY (-1);
418 				surface.setHighlightZ (-1);
419 			}
420 		}
421 		// redraw correctly the scene
422 		this.show (true);
423 	}
424 };

442 /**
443  * @override
444  * Moves the camera.
445  *
446  * @param {MouseEvent} event - The event.
447  *
448  * @return {void}
449  */
450 SurfaceViewer.prototype.onMouseMove = function (event) {
451 	if ((event.buttons == 4 || (event.buttons == 1 && event.ctrlKey)) 
452 		&& this.mousePosOnPress[0] != -1) 
453 	{
454 		/* middle button is pressed (or left button and ctrl key pressed) and
455 		 * the user generate mousedown event on the surface canvas.
456 		 */
457 		this.moveCamera (
458 			(event.layerX - this.mousePosOnPress[0]) * 0.005,
459 			(event.layerY - this.mousePosOnPress[1]) * 0.005
460 		);
461 	}
462 	else if (event.buttons == 1 && this.mousePosOnPress[0] != -1) {
463 		/* left button is pressed and the user generate mousedown event on the
464 		 * surface canvas.
465 		 */
466 		this.rotateCamera (
467 			(this.mousePosOnPress[0] - event.layerX) * 0.01,
468 			(event.layerY - this.mousePosOnPress[1]) * 0.01
469 		);
470 	}
471 };

537 /**
538  * Moves the camera using spheric coordinates.
539  *
540  * @param {Number} phiOffset - Azimuth offset.
541  * @param {Number} thetaOffset - Altitude offset.
542  *
543  * @return {void}
544  * @throw {String} the parameters are not of type number
545  */
546 SurfaceViewer.prototype.rotateCamera = function (phiOffset, thetaOffset) {
547 	/// parameter verification
548 	if (! checkType (arguments, "number", "number")) {
549 		throw "SurfaceViewer.rotateCamera: one of the parameters is not a number";
550 	}
551 
552 	/// compute angle
553 	var pos = new Vector (this.camPosWhenClick).sub (this.camCenterWhenClick);
554 	var dist = pos.getLength ();
555 	var phi = getAzimuth (this.camPosWhenClick, this.camCenterWhenClick) + phiOffset;
556 	var theta = clamp (
557 		-Math.PI / 2 + 0.0002, 
558 		Math.PI / 2 - 0.0002, 
559 		getAltitude (
560 			this.camPosWhenClick, 
561 			this.camCenterWhenClick
562 		) + thetaOffset
563 	);
564 
565 
566 	/// compute position
567 	pos.x = dist * Math.cos (phi) * Math.cos (theta);
568 	pos.y = dist * Math.sin (phi) * Math.cos (theta);
569 	pos.z = dist * Math.sin (theta);
570 	this.container.setCameraAt (pos.add (this.camCenterWhenClick));
571 
572 	/// drawing
573 	this.draw ();
574 };

578 /**
579  * Translates the camera (eye and center position).
580  *
581  * @param {Number} xOffset - X offset (left-right).
582  * @param {Number} yOffset - Y offset (up-down).
583  *
584  * @return {void}
585  * @throw {String} the parameters are not of type number
586  */
587 SurfaceViewer.prototype.moveCamera = function (xOffset, yOffset) {
588 	/// parameter verification
589 	if (! checkType (arguments, "number", "number")) {
590 		throw "SurfaceViewer.moveCamera: one of parameter are not number";
591 	}
592 	
593 	/// compute local base
594 	var base = new Matrix ()
595 		.rotateZ (getAzimuth (this.camPosWhenClick, this.camCenterWhenClick))
596 	base.rotate (
597 		getAltitude (this.camPosWhenClick, this.camCenterWhenClick),
598 		base.getYVector ()
599 	);
600 	
601 	/// compute translation
602 	var t = new Vector (base.getYVector ())
603 		.mul (-xOffset)
604 		.add (new Vector (base.getZVector ()).mul (yOffset));
605 	
606 	/// set camera position
607 	this.container.setCameraAt (
608 		addVector (this.camPosWhenClick, t),
609 		addVector (this.camCenterWhenClick, t)
610 	);
611 
612 	/// drawing
613 	this.draw ();
614 };

642 /**
643  * @param {int} width - The width of the canvas from which we get the data.
644  * @param {int} height - The height of the canvas from which we get the data.
645  * 
646  * @return {float[]} The image data
647  */
648 SurfaceViewer.prototype.getImgData = function (width, height) {
649 	// Bind the frame framebuffer and the depth buffer for the color rendering
650 	this.glContext.bindRenderbuffer (this.glContext.RENDERBUFFER,
651 		this.screenBuffer);
652 	this.glContext.bindFramebuffer (this.glContext.FRAMEBUFFER,
653 		this.framebuffer);
654 
655 	// Drawing the colored scene
656 	this.draw ();
657 
658 	// Pixel on which we click
659 	var pixel = new Uint8Array (width * height * 4);
660 
661 	// Read the pixel at x and y coordinates
662 	this.glContext.readPixels (0, 0,
663 		width, height,
664 		this.glContext.RGBA, this.glContext.UNSIGNED_BYTE,
665 		pixel
666 	);
667 	// Unbind the buffers used
668 	this.glContext.bindRenderbuffer (this.glContext.RENDERBUFFER, null);
669 	this.glContext.bindFramebuffer (this.glContext.FRAMEBUFFER, null);
670 
671 	return this.reverseTab (pixel, width, height);
672 };

676 /**
677  * Vertical flip of a matrix.
678  * @example | 1 2 |  ->  | 3 4 |
679  *          | 3 4 |      | 1 2 |
680  * XXX Function instead of method?
681  * 
682  * @param {Array} tab - An array (like a matrix).
683  * @param {int} width - Matrix width.
684  * @param {int} height - Matrix height.
685  * 
686  * @return {Array} The flipped matrix.
687  */
688 SurfaceViewer.prototype.reverseTab = function (tab, width, height) {
689 	var pixels = [];
690 	for (var i = height - 1; i >= 0; --i) {
691 		for (var j = 0; j < width * 4; ++j) {
692 			pixels.push (tab[i * width * 4 + j]);
693 		}
694 	}
695 	return pixels;
696 };
