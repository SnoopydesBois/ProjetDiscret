 90 /**
 91  * @constructor
 92  *
 93  * @param {Range} image - The image range of the curve.
 94  * @param {Range} inverseImage - The inverse image range of the curve.
 95  */
 96 function ModelCurve (image, inverseImage) {
 97 	if (! checkType (arguments, Range, Range)) {
 98 		console.error ("ModelCurve.constructor: bad type(s) of parameter(s)");
 99 	}
100 
101 	/**
102 	 * {Range} The image range of the curve.
103 	 */
104 	this.image = image;
105 
106 	/**
107 	 * {Range} The inverse image range of the curve.
108  	 */
109 	this.inverseImage = inverseImage;
110 
111 	/**
112 	 * {Map<String, Function>} The list of curves available for the user
113 	 * Contains the default list of curves.
114 	 */
115 	this.listCurve = new Map ();
116 
117 	/**
118 	 * {Curve} The active curve of the model.
119 	 */
120 	this.activeCurve = null;
121 };

131 /**
132  * Adds a curve to its list.
133  *
134  * @param {String} name - The name of the curve.

136  * constructor must inherit from Curve.

141 	this.listCurve.set (name, constructor);

145 //==============================================================================
146 /**
147  * Sets the new active curve for the model.
148  * If the type of the curve is provided, a new curve is created with
149  * the parameter curve as it's equation.
150  * If the type of the curve is not provided, the new active curve is a
151  * default curve and the parameter "curve" is the name of this default curve
152  * present in the attribute listCurve of the model (i.e the key of the Map).
153  *
154  * @param {(Equation | String)} curve - The equation of the curve or its name.
155  * @param {EquationTypeEnum} [type] - The type of the new active curve.
156  *
157  * @return {void}
158  */
159 ModelCurve.prototype.setActive = function (curve, type) {
160 	/// paramters verification
161 	if (! checkType (arguments, ["string", Equation])) {
162 		console.error("ModelCurve.setActive: bad type of 'curve' parameter");
163 	}
164 
165 	/// set
166 	switch (type) {
167 		case EquationTypeEnum.IMPLICIT :
168 			this.activeCurve = new ImplicitCurve (curve);
169 			break;
170 		case EquationTypeEnum.EXPLICIT :
171 			this.activeCurve = new ExplicitCurve (curve);
172 			break;
173 		case EquationTypeEnum.PARAMETRIC :
174 			throw "ModelCurve.setActive: This model does not handle parametric "
175 			 + "curves";
176 			break;
177 		case undefined:
178 		case null:
179 			if (this.listCurve.has (curve)) {
180 				this.activeCurve = new (this.listCurve.get(curve))();
181 				if (this.activeCurve instanceof ExplicitCurve) {
182 					this.image = this.activeCurve.computeRange ();
183 					this.inverseImage = this.activeCurve.computeRange ();
184 				}
185 			}
186 			else {
187 				this.consolePrintAvailableCurve ();
188 				throw "ModelCurve.setActive: Unknown equation name";
189 			}
190 			break;
191 		default :
192 			throw "ModelCurve.setActive: Unknown type of equation";
193 			break;
194 	}
195 	return this.activeCurve;
196 };

202  */
203 ModelCurve.prototype.getPoints = function () {
204 	return this.activeCurve.computePoints (this.image, this.inverseImage, 0.1);
205 };

297 /**
298  * @param {Equation} eq - The equation to set to the active implicit/explicit
299  * curve.
300  * @return {void}
301  * @throw {String} The parameter is not of type Equation.
302  */
303 ModelCurve.prototype.addEquation = function (eq) {
304 	if (!(eq instanceof Equation)) {
305 		throw "ModelCurve.addEquation: eq parameter is of wrong type";
306 	}
307 	this.activeCurve.setEquation (eq);
308 };
