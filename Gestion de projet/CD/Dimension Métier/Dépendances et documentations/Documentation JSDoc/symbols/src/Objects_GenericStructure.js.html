 95 /**
 96  * @constructor
 97  * 
 98  * @param {String} name - The name of the structure.
 99  * @param {Shader} shader - The shader used to draw the structure.
100  */
101 function GenericStructure (name, shader) {
102 	
103 	if (arguments.length != 0 // GenericStructure is a super class.
104 		&& !(typeof name == "string"
105 		&& shader instanceof Shader)) 
106 	{
107 		console.error ("GenericStructure.constructor: bad type(s) of " + 
108 			"parameter(s)");
109 		showType (name, shader);
110 		return;
111 	}
112 	
113 	/**
114 	 * {String} Name of the object to use.
115 	 */
116 	this.structureName = name;
117 	
118 	/**
119 	 * {Matrix} Current object matrix.
120 	 */
121 	this.matrix = new Matrix ();
122 	
123 	/**
124 	 * {Shader} Shader to use to render the current object.
125 	 */
126 	this.shader = shader; 
127 	
128 	/**
129 	 * {boolean} Indicates if the structure is displayable. True by default.
130 	 */
131 	this.isDisplayable = true;
132 	
133 	/**
134 	 * {boolean} Indicates if the structure is pickable. True by default.
135 	 */
136 	this.isPickable = true;
137 	
138 	/**
139 	 * {int} Number of needed gl buffer to draw the object. Set by 'prepare'
140 	 * method, use by 'prepare' and 'draw' methods.
141 	 */
142 	this.nbGlBuffer = 0;
143 	
144 	/**
145 	 * {WebGLBuffer[]} Array of gl vertex buffer. Filled by 'prepare' method. The length of
146 	 * this array is 'this.nbGlBuffer'.
147 	 */
148 	this.glVertexBuffer = [];
149 
150 	/**
151 	 * {WebGLBuffer[]} Array of gl vertex color buffer for picking. Filled by 'prepare'
152 	 * method. The length of this array is 'this.nbGlBuffer'.
153 	 */
154 	this.glBackBuffer = [];
155 
156 	/**
157 	 * {WebGLBuffer[]} Array of gl indices buffer. Filled by 'prepare' method. The length
158 	 * of this array is 'this.nbGlBuffer'.
159 	 */
160 	this.glIndiciesBuffer = [];
161 	
162 	/**
163 	 * {boolean} This value indicates if the structure is already prepared.
164 	 */
165 	this.prepared = false;
166 };

403 /**
404  * Adds a position into an array of data.
405  * 
406  * @param {Array} data - The array to push position's coordinates into.
407  * @param {(float | float[3])} X - The X coordinate (if it is a float value) or
408  * all coordinates (if it is an array) of the point.
409  * @param {float} [Y] - The Y coordinate of the point. Mandatory if X is a float
410  * value.
411  * @param {float} [Z] - The Z coordinate of the point. Mandatory if X is a float
412  * value.
413  * 
414  * @return {int} 0 if the shader doesn't accept position, 3 otherwise.
415  * @throw {String} Some parameters are missing.
416  */
417 GenericStructure.prototype.addAPoint = function (data, X, Y, Z) {
418 	if (! this.shader.hasAttribute (AttributeEnum.position)) {
419 		return 0;
420 	}
421 	
422 	if (X instanceof Array) { // If X is an Array
423 		data.push (X[0], X[1], X[2]);
424 	}
425 	else if (Z === undefined) { // If one parameter is missing
426 		throw "GenericObject.addAPoint: missing parameter (typeof X: "
427 				+ typeof X + ")";
428 	}
429 	else { // X, Y, Z are 3 coordinates
430 		data.push (X, Y, Z);
431 	}
432 	
433 	return 3;
434 };

438 /**
439  * Adds a color into an array of data.
440  * 
441  * @param {Array} data - The array to push color's component into.
442  * @param {(flaot | float[4])} R - The Red value (if it is a float value) or the
443  * four color component (if it is an array) to push into data.
444  * @param {float} [G] - The Green value to push into data. Mandatory if R is a
445  * float value.
446  * @param {float} [B] - The Blue value to push into data. Mandatory if R is a
447  * float value.
448  * @param {float} [A] - The Alpha value to push into data. Mandatory if R is a
449  * float value.
450  * 
451  * @return {int} 0 if the shader doesn't accept color, 4 otherwise.
452  * @throw {String} Some parameters are missing.
453  */
454 GenericStructure.prototype.addAColor = function (data, R, G, B, A) {
455 	if (! this.shader.hasAttribute (AttributeEnum.color)) {
456 		return 0;
457 	}
458 	
459 	if (R instanceof Array) { // If R is an array
460 		data.push (R[0], R[1], R[2], R[3]);
461 	}
462 	else if (A === undefined) { // If one parameter is missing
463 		throw "GenericObject.addAColor: missing parameter (typeof R: "
464 				+ typeof R + ")";
465 	}
466 	else { // R, G, B, A are 4 values
467 		data.push (R, G, B, A);
468 	}
469 	return 4;
470 };

578  *

581  * value) or all coordinates (if it is an array) of the texture coordinates.
582  * @param {float} [V] - The V texture coordinate. Mandatory if X is a float
583  * value.
584  *

586  * @param {String} Some parameters are missing.
587  */
588 GenericStructure.prototype.addTextureCoordinates = function (data, U, V) {
589 	if (! this.shader.hasAttribute (AttributeEnum.texcoord)) {
590 		return 0;
591 	}
592 	
593 	if (U instanceof Array) { // If U is an array
594 		data.push (U[0], U[1]);
595 	}
596 	else if (V === undefined) { // If one parameter is missing
597 		throw "GenericObject.addTextureCoordinates: missing parameter " +
598 				+ "(typeof U: " + typeof U + ")";
599 	}
600 	else { // U, V are 2 values
601 		data.push (U, V);
602 	}
603 	return 2;
