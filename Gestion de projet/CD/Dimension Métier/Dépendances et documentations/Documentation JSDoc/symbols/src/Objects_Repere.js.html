 81 /**
 82  * @constructor
 83  *
 84  * @param {WebGLRenderingContext} glContext - The gl context (used by the
 85  * shader).
 86  */
 87 function Repere (glContext) {
 88 	GenericStructure.call (this, "repere", new DefaultShader (glContext));
 89 
 90 	/**
 91 	 * {Vector} The dimension of the 3D space in voxel.
 92 	 */
 93 	this.dimension = new Vector (1, 1, 1);
 94 
 95 	/**
 96 	 * @inheritdoc
 97 	 * An user cannot select the repere.
 98 	 */
 99 	this.isPickable = false;
100 
101 
102 	this.matrix.scale (0.01);
103 };

122 /**
123  * @override
124  * Computes and gets the model matrix. The repere is translated to the image
125  * bottom-left corner to the given camera.
126  *
127  * @param {Camera} camera - The active camera.
128  *
129  * @return {Matrix} The model matrix.
130  * @throws {String} The provided parameter is not of type Camera.
131  */
132 Repere.prototype.getMatrix = function (camera) {
133 	/// parameter verification
134 	if (! checkType (arguments, Camera))
135 		throw "Repere.getMatrix: Given parameter is not a Camera";
136 
137 	var base = new Matrix ()
138 		.rotateZ (getAzimuth (
139 			camera.getPosition (), 
140 			camera.getLookAtPosition ()
141 		))
142 	base.rotate (
143 		getAltitude (camera.getPosition (), camera.getLookAtPosition ()),
144 		base.getYVector ()
145 	);
146 	/// compute and return the matrix
147 	var coef = 0.083 / camera.height;
148 //	var coef = 2.0 / camera.height;
149 	var t = new Vector (camera.getPosition ())
150 		.sub (camera.getLookAtPosition ())
151 		.normalize ();
152 	var ty = base.getYVector ().mul (- camera.width * coef);
153 	var tz = base.getZVector ().mul (- camera.height * coef)
154 	t = new Vector (camera.getPosition ())
155 		.sub (t.mul (0.2))
156 		.add (ty).sub (new Vector (ty).mul (0.12))
157 		.add (tz).sub (new Vector (tz).mul (0.12));
158 	
159 	return new Matrix (this.matrix).translate (t);
160 };

170 /**
171  * @override
172  *
173  * @param {WebGLRenderingContext} glContext - The gl context.
174  * 
175  * @return {void}
176  */
177 Repere.prototype.prepare = function (glContext) {
178 	/// Vertex and color buffers
179 	var vertexBuffer = [ // the vertex of the box which serves as repere
180 		// X axis
181 		[0.1,  0.0   ,  0.095 ],
182 		[0.7,  0.0   ,  0.095 ],
183 		[0.7, -0.1   ,  0.195 ],
184 		[1.0,  0.0495,  0.0495],
185 		[0.7,  0.195 , -0.1   ],
186 		[0.7,  0.095 ,  0.0   ],
187 		[0.1,  0.095 ,  0.0   ],
188 		// Y axis
189 		[ 0.095 , 0.1,  0.0   ],
190 		[ 0.095 , 0.7,  0.0   ],
191 		[ 0.195 , 0.7, -0.1   ],
192 		[ 0.0495, 1.0,  0.0495],
193 		[-0.1   , 0.7,  0.195 ],
194 		[ 0.0   , 0.7,  0.095 ],
195 		[ 0.0   , 0.1,  0.095 ],
196 		// Z axis
197 		[ 0.0   ,  0.095 , 0.1],
198 		[ 0.0   ,  0.095 , 0.7],
199 		[-0.1   ,  0.195 , 0.7],
200 		[ 0.0495,  0.0495, 1.0],
201 		[ 0.195 , -0.1   , 0.7],
202 		[ 0.095 ,  0.0   , 0.7],
203 		[ 0.095 ,  0.0   , 0.1]
204 	];
205 
206 	var vertexBufferLength = vertexBuffer.length; // 8
207 	// Color of each vertex, lines are white
208 
209 
210 	/// Vertex Buffer
211 	this.glVertexBuffer = glContext.createBuffer ();
212 	this.glVertexBuffer.numItems = vertexBufferLength;
213 
214 	var data = [];
215 	for (var vertice = 0; vertice < vertexBufferLength; ++vertice) {
216 		this.addAPoint (data, vertexBuffer[vertice]);
217 		if (vertice < 7)
218 			this.addAColor (data, [0.9, 0.0, 0.0, 1.0]);
219 		else if (vertice < 14)
220 			this.addAColor (data, [0.0, 0.9, 0.0, 1.0]);
221 		else
222 			this.addAColor (data, [0.0, 0.0, 0.9, 1.0]);
223 	}
224 	glContext.bindBuffer (glContext.ARRAY_BUFFER, this.glVertexBuffer);
225 	glContext.bufferData (glContext.ARRAY_BUFFER, new Float32Array (data),
226 		glContext.STATIC_DRAW);
227 
228 
229 	/// Finish, tell it
230 	this.prepared = true;
231 };

235 /**
236  * @override
237  *
238  * @param {WebGLRenderingContext} glContext - The gl context.
239  *
240  * @return {void}
241  */
242 Repere.prototype.draw = function (glContext) {
243 	/// Parameter verification
244 	if (!(glContext instanceof WebGLRenderingContext)) {
245 		console.error ("Repere.draw: glContext is not a WebGLRenderingContext");
246 		return;
247 	}
248 
249 	/// Buffers verification
250 	if (this.glVertexBuffer === undefined) {
251 		console.error ("Repere.draw: prepare the repere BEFORE drawing it !");
252 		return;
253 	}
254 
255 	// Let's the shader prepare its attributes
256 	this.shader.setAttributes (glContext, this.glVertexBuffer, 1);
257 
258 	// Let's render !
259 	glContext.drawArrays (glContext.LINE_LOOP, 0, this.glVertexBuffer.numItems);
260 };
