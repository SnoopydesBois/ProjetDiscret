 88 /**
 89  * @constructor
 90  * 
 91  * @param {Vector} size - Vector to define the size of the model.
 92  */
 93 function Surface (size) {
 94 	/// parameter verification
 95 	if (! size instanceof Vector) {
 96 		console.error ("Surface.constructor: parameter is not a Vector");
 97 	}
 98 	
 99 	/**
100 	 * {Vector} The dimensions of the surface
101 	 */
102 	this.dimension = new Vector (size);
103 	
104 	/**
105 	 * {Array} 3 dimensional array containing the surface data.
106 	 */
107 	this.matVoxel = [];
108 	for (var x = 0; x < this.dimension.x; ++x) {
109 		this.matVoxel[x] = [];
110 		for (var y = 0; y < this.dimension.y; ++y) {
111 			this.matVoxel[x][y] = [];
112 			for (var z = 0; z < this.dimension.z; ++z) {
113 				this.matVoxel[x][y][z] = null;
114 			}
115 		}
116 	}
117 	
118 	/**
119 	 * {int} Number of voxels the surface contains
120 	 */
121 	this.nbVoxel = 0;
122 };

132 /**
133  * @param {Vector | float[3] | float } position - the voxel's coordinates.
134  * @param {float} [yCoord] - The y coordinate.
135  * @param {float} [zCoord] - The z coordinate.
136  * 
137  * @return {Voxel} voxel at the specified coordinates.
138  * @throws {String} "Surface.getVoxel.ErrorNotAVector"
139  * - Position should be of type Vector.
140  */
141 Surface.prototype.getVoxel = function (position, yCoord, zCoord) {
142 	var x, y, z;
143 	if (position instanceof Vector) {
144 		x = position.x;
145 		y = position.y;
146 		z = position.z;
147 	}
148 	else if (position instanceof Array && position.length >= 3) {
149 		x = position[0];
150 		y = position[1];
151 		z = position[2];
152 	}
153 	else if (checkType (arguments, "number", "number", "number")) {
154 		x = position;
155 		y = yCoord;
156 		z = zCoord;
157 	}
158 	else
159 		throw "Surface.getVoxel: bad type(s) of parameter(s)"
160 	
161 	return (this.isIn (x, y, z)) ? this.matVoxel[x][y][z] : null;
162 };

166 /**
167  * Adds a voxel to the model.
168  * @param {Vector} position - The coordinates of the voxel to add.
169  * @param {EnumConnexity} connexity - The connexity for which the voxel should 
170  * be displayed.
171  * 
172  * @throws {String} The parameters are not of expected type.
173  * @throws {String} "Surface.addVoxel.OutOfBounds"
174  * - the voxel is out of bounds
175  */
176 Surface.prototype.addVoxel = function (position, connexity) {
177 	/// parameters verification
178 	if (! checkType (arguments, Vector, "number"))
179 		throw "Surface.addVoxel: bad type(s) of parameter(s)";
180 	
181 	/// add the voxel
182 	var x = position.x;
183 	var y = position.y;
184 	var z = position.z;
185 	
186 	if (this.isIn (x, y, z)) {
187 		if (this.matVoxel[x][y][z] === null) {
188 			
189 			this.matVoxel[x][y][z] = new Voxel (position, connexity);
190 			this.nbVoxel++;
191 		}
192 		else {
193 			// the voxel already exists, just add connexity
194 			this.matVoxel[x][y][z].connexity |= connexity;
195 		}
196 		var size = DirectionEnum.size;
197 		for (var i = 0; i < size; ++i) {
198 			var newX = x + DirectionEnum.properties[i].x;
199 			var newY = y + DirectionEnum.properties[i].y;
200 			var newZ = z + DirectionEnum.properties[i].z;
201 			
202 			if (this.isIn (newX, newY, newZ) 
203 				&& this.matVoxel[newX][newY][newZ] !== null)
204 			{
205 				this.matVoxel[x][y][z].addFacetConnexity (
206 					i, 
207 					this.matVoxel[newX][newY][newZ].getConnexity ()
208 				);
209 				this.matVoxel[newX][newY][newZ].addFacetConnexity (
210 					DirectionEnum.properties[i].oppose,
211 					connexity
212 				);
213 			}
214 		} // end for each neighbor
215 	} // end if coordinates are corect
216 	else {
217 		console.error ("out of bound:", x, y, z);
218 		throw "Surface.addVoxel.OutOfBounds";
219 	}
220 };

224 /**
225  * Tests if a voxel has a visible facet
226  * 
227  * @param {Vector} voxelPosition - The coordinates of the voxel to test.
228  * @param {DirectionEnum} direction - The direction of the facet.
229  * @param {ConnexityEnum} connexity - The current connexity.
230  * 
231  * @return {boolean} True if the voxel is visible, false otherwise.
232  */
233 Surface.prototype.voxelHasFacet = function (voxelPosition, direction, 
234 	connexity)
235 {
236 	var voxel = this.getVoxel (voxelPosition);
237 	var nx = voxelPosition.x + DirectionEnum.properties[direction].x,
238 		ny = voxelPosition.y + DirectionEnum.properties[direction].y,
239 		nz = voxelPosition.z + DirectionEnum.properties[direction].z;
240 	var neighbor = (this.isIn (nx, ny, nz)) ? this.getVoxel (nx, ny, nz) : null;
241 	return (neighbor == null || !neighbor.visibility ||
242 		(voxel.getConnexity () < neighbor.getConnexity ()
243 		&& neighbor.getConnexity () > connexity)
244 	);
245 };

249 /**
250  * @param {int} x - x-coordinates.
251  * @param {int} y - y-coordinates.
252  * @param {int} z - z-coordinates.
253  * 
254  * @return {boolean} true if the coordinates are in the matrix, false otherwise.
255  * @throws {String} "Surface.isIn.ErrorNotANumber"
256  * - the coordinates should be numbers
257  */
258 Surface.prototype.isIn = function (x, y, z) {
259 	if (! checkType (arguments, "number", "number", "number"))
260 		throw "Surface.isIn.ErrorNotANumber";
261 	
262 	return (x >= 0 && x < this.dimension.x &&
263 		y >= 0 && y < this.dimension.y &&
264 		z >= 0 && z < this.dimension.z);
265 };

279  * Empties the matrix.

282  */
283 Surface.prototype.clear = function () {
284 	for (var x = 0; x < this.dimension.x; ++x) {
285 		for (var y = 0; y < this.dimension.y; ++y) {
286 			for (var z = 0; z < this.dimension.z; ++z) {
287 				this.matVoxel[x][y][z] = null;
288 			}
289 		}
290 	}
291 	this.nbVoxel = 0;
292 };

296 /**
297  * Gets the numbers of neighbors of a voxel.
298  *
299  * @param {Vector} position - the coordinates of the voxel.
300  *
301  * @return {int} number of neighbours.
302  * @throws {String} "Surface.getNbNeighbor.ErrorNotAVector"
303  * - the position should be of type Vector
304  */
305 Surface.prototype.getNbNeighbor = function (position) {
306 	if (!position instanceof Vector)
307 		throw "Surface.getNbNeighbor.ErrorNotAVector";
308 	
309 	var nb = 0; // number of neighbours
310 	var size = DirectionEnum.size;
311 	for (var i = 0; i < size; ++i) {
312 		var x = position.x + DirectionEnum.properties[i].x;
313 		var y = position.y + DirectionEnum.properties[i].y;
314 		var z = position.z + DirectionEnum.properties[i].z;
315 
316 		if (this.isIn (x, y, z) && this.matVoxel[x][y][z] !== null)
317 			++nb;
318 	}
319 	return nb;
320 };

361  * @param {float} [yCoord] - The y-coordinate.

367  * @throws {String} "Surface.isVoxelVisible.OutOfBounds"
368  * - the voxel is out of bounds.

370 Surface.prototype.isVoxelVisible = function (position, yCoord, zCoord) {
371 	/// coordinates assignation
372 	var x, y, z;
373 	if (position instanceof Vector) {
374 		x = position.x;
375 		y = position.y;
376 		z = position.z;

378 	else if (position instanceof Array && position.length >= 3) {
379 		x = position[0];

400 /**
401  * Sets the voxels out of the new range to not visible so they are not displayed.
402  * 
403  * @param {Range} range - The new range of the slice.
404  * @param {AxisEnum} axis - The axis on which the slice is performed.
405  * 
406  * @return {void}
407  * @throws {String} "Surface.slice.ErrorNotARange" 
408  * - The range should be of type Range.
409  */
410 Surface.prototype.slice = function (range, axis) {
411 	if (! (range instanceof Range))
412 		throw "Surface.slice.ErrorNotARange";
413 	
414 	var visible, x, y, z, i, newX, newY, newZ;
415 	for (x = 0; x < this.dimension.x; ++x) {
416 		for (y = 0; y < this.dimension.y; ++y) {			
417 			for (z = 0; z < this.dimension.z; ++z) {
418 				switch (axis) {
419 					case AxisEnum.X :
420 						visible = range.isIn (x);
421 					break;
422 					case AxisEnum.Y :
423 						visible = range.isIn (y);
424 					break;
425 					case AxisEnum.Z :
426 						visible = range.isIn (z);
427 					break;
428 					default :
429 						visible = false;
430 					break;
431 				}
432 				if (this.matVoxel[x][y][z] !== null) {
433 					this.matVoxel[x][y][z].visibility = visible;
434 					for (i = 0; i < DirectionEnum.size; ++i) {
435 						newX = x + DirectionEnum.properties[i].x;
436 						newY = y + DirectionEnum.properties[i].y;
437 						newZ = z + DirectionEnum.properties[i].z;
438 			
439 						if (this.isIn (newX, newY, newZ) 
440 							&& this.matVoxel[newX][newY][newZ] !== null)
441 						{
442 							this.matVoxel[newX][newY][newZ]
443 								.neighborVisibility[DirectionEnum.properties[i].oppose] = visible;
444 						}
445 					} // end for each neighbor
446 				}
447 			} // end for z
448 		} // end for y
449 	} // end for x
450 };

454 /**
455  * @see {@link slice}. Perform a slice on 3 axis
456  * 
457  * @param {Range} rangeX - The new x-Range.
458  * @param {Range} rangeY - The new y-Range.
459  * @param {Range} rangeZ - The new z-Range.
460  * 
461  * @return {void}
462  * @throws {String} "Surface.printOnly.ErrorNotARange" 
463  * - All ranges should be of type Range.
464  */
465 Surface.prototype.printOnly = function (rangeX, rangeY, rangeZ) {
466 	if (! checkType (arguments, Range, Range, Range))
467 		throw "Surface.printOnly.ErrorNotARange";
468 	
469 	var visible, x, y, z, i, newX, newY, newZ;
470 	for (x = 0; x < this.dimension.x; ++x) {
471 		for (y = 0; y < this.dimension.y; ++y) {			
472 			for (z = 0; z < this.dimension.z; ++z) {
473 				voxel = this.matVoxel[x][y][z];
474 				if (voxel !== null) {
475 					visible = rangeX.isIn (x) && rangeY.isIn (y)
476 						&& rangeZ.isIn (z);
477 					voxel.visibility = visible;
478 					for (i = 0; i < DirectionEnum.size; ++i) {
479 						newX = x + DirectionEnum.properties[i].x;
480 						newY = y + DirectionEnum.properties[i].y;
481 						newZ = z + DirectionEnum.properties[i].z;
482 			
483 						if (this.isIn (newX, newY, newZ) 
484 							&& this.matVoxel[newX][newY][newZ] !== null)
485 						{
486 							this.matVoxel[newX][newY][newZ]
487 								.neighborVisibility[
488 									DirectionEnum.properties[i].oppose
489 								] = visible;
490 						}
491 					} // end for each neighbor
492 				} // end if voxel exist
493 			} // end for z
494 		} // end for y
495 	} // end for x
496 };

500 /**
501  * Sets a new connexity to the Surface.
502  * 
503  * @param {ConnexityEnum} connexity - The new connexity of the surface.
504  * 
505  * @return {void}
506  * @throws {String} "Surface.setConnexity.ErrorNotAConnexity" - Connexity should
507  * be a ConnexityEnum.
508  */
509 Surface.prototype.setConnexity = function (connexity) {
510 	if (! isValueOfEnum (ConnexityEnum, connexity))
511 		throw "Surface.setConnexity.ErrorNotAConnexity";
512 	
513 	var vox;
514 	for (var x = 0; x < this.dimension.x; ++x) {
515 		for (var y = 0; y < this.dimension.y; ++y) {			
516 			for (var z = 0; z < this.dimension.z; ++z) {
517 				vox = this.matVoxel[x][y][z];
518 				if (vox != null && vox.connexity | connexity){
519 					vox.visibility = true;
520 				}
521 			} // end for each z
522 		} // end for each y
523 	} // end for each x
524 };
