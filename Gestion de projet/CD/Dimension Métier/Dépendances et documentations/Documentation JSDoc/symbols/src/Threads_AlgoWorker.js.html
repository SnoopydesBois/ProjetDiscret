 74 /**
 75  * @constructor 
 76  * A AlgoWorker contains a number of workers that will compute the surface.
 77  * This class is abstract and the algorithm used shall be defined in the
 78  * subclass.
 79  * 
 80  * @param {(Equation | DrawnCurve)} meridianCurve - The equation for the
 81  * meridian.
 82  * @param {Equation} revolutionCurve - The equation for the revolution curve.
 83  * @param {Vector} dimension - The space dimensions.
 84  * @param {Surface} surface - The surface to draw.
 85  */
 86 function AlgoWorker (meridianCurve, revolutionCurve, dimension, surface) {
 87 	/**
 88 	 * {Surface} The surface to draw.
 89 	 */
 90 	this.surface = surface;
 91 	
 92 	/**
 93 	 * {boolean} Whether the algorithm has finished.
 94 	 */
 95 	this.finished = false;
 96 	
 97 	/**
 98 	 * {int} Nb of workers that are still active.
 99 	 */
100 	this.activeWorkers = 0;
101 	
102 	/**
103 	 * {Array} TODO FIXME transformer type en "type[]".
104 	 */
105 	this.worker = [];
106 	
107 	/**
108 	 * {(Stirng | Number[][2])} TODO
109 	 */
110 	this.meridianCurve = null;
111 	if (meridianCurve instanceof Equation) {
112 		this.meridianCurve = meridianCurve.toStringNoParam ();
113 	}
114 	else if (meridianCurve instanceof DrawnCurve) {
115 		this.meridianCurve = [meridianCurve.xList, meridianCurve.yList];
116 	}
117 	
118 	this.revolutionCurve = 
119 		(revolutionCurve ? revolutionCurve.toStringNoParam() : "");
120 	
121 	/**
122 	 * {float[3]} The dimension of the 3D space.
123 	 */
124 	this.dim = (dimension ? dimension.m : null);
125 };

135 /**
136  * This method add the voxels contained in a buffer every time a new message
137  * is received from a worker.
138  * 
139  * @param {Number[4][]} buffer - A buffer containing voxel. Each voxel is an
140  * array: [x, y, z, connexity]
141  * @param {int} size - Size of the buffer.
142  * 
143  * @return {void}
144  */
145 AlgoWorker.prototype.readBuffer = function (buffer, size) {
146 	for (var i = 0; i < size; ++i){
147 		var voxel = buffer[i];
148 		try {
149 			this.surface.addVoxel (new Vector (voxel[0], voxel[1], voxel[2]),
150 				voxel[3]);
151 		}
152 		catch (e) {
153 			appli.alertMessage ("Somevoxels were out of the bounds and will "
154 				+ "not be dispalyed");
155 			console.log ("Some voxels are out of the bounds");
156 		}
157 	}
158 };
