110 /**
111  * This function generate the surface using the algorithm for explicit
112  * functions whith no optimisation.
113  * 
114  * @return {void}
115  */
116 function algo () {
117 	var dimx = dimension.x;
118 	var dimy = dimension.y;
119 	var dimz = dimension.z;
120 	var maxx = Math.trunc (dimx / 2);
121 	var maxy = Math.trunc (dimy / 2);
122 	zMax = Math.min (zMax, dimz);
123 	for (var z = zMin; z < zMax; ++z) {
124 		var rz = explicit_curve.compute ([z]);
125 		if (rz == 0)
126 			rz += 0.01;
127 		var rz1 = explicit_curve.compute ([z - 0.5]);
128 		if (rz1 == 0)
129 			rz1 += 0.01;
130 		var rz2 = explicit_curve.compute ([z + 0.5]);
131 		if (rz2 == 0)
132 			rz2 += 0.01;
133 		var buffer = [];
134 		var bufferSize = 0;
135 		for (var y = 0; y < dimy; y++) {
136 			for (var x = 0; x < dimx; x++) {
137 				connexity = checkVoxel (implicit_curve, x - maxx, y - maxy,
138 					[rz, rz1, rz2]);
139 				if (connexity != 0) {
140 					buffer.push([x,y,z,connexity]);
141 					bufferSize ++;
142 				}
143 			} // end for x
144 		} // end for y
145 		if (bufferSize !=0)
146 			postMessage ([buffer, bufferSize]);
147 	} // end for z
148 }

152 /**
153  * Receive the meridian, revolution curve and dimension, then launch algorithm
154  * and ask for termination.
155  * 
156  * @param {Event} e - e.data contains the message received.
157  * 
158  * @return {void}
159  */
160 onmessage = function (e) {
161 	id = e.data[0];
162 	explicit_curve = new Equation (e.data[1]);
163 	implicit_curve = new Equation (e.data[2]);
164 	dimension = new Vector (e.data[3]);
165 	zMin = e.data[4];
166 	zMax = e.data[5];
167 	try {
168 		algo ();
169 	}
170 	catch (e) {
171 		postMessage (["Abort"]);
172 	}
173 	postMessage (["Terminate", id]);
174 };
