116 /**
117  * This function generate the surface using the algorithm for explicit
118  * functions.
119  * 
120  * @return {void}
121  */
122 function algo () {
123 	var maxx = Math.trunc (dimx / 2);
124 	var maxy = Math.trunc (dimy / 2);
125 	var buffer = [];
126 	var bufferSize = 0;
127 	var tValues = [];
128 	var connexity = "";
129 	var x, y, t, z;
130 	//find first voxel
131 	for (t = 0; t < dimt && bufferSize === 0; ++z) {
132 		tValues[t] = [];
133 		tValues[t][0] = parametric_curve.getX (t);
134 		if (tValues[t][0] === 0)
135 			tValues[t][0] = 0.001;
136 		tValues[t][1] = parametric_curve.getX (t - 0.5);
137 		if (tValues[t][1] === 0)
138 			tValues[t][1] = 0.001;
139 		tValues[t][2] = parametric_curve.getX (t + 0.5);
140 		if (tValues[t][2] === 0)
141 			tValues[t][2] = 0.001;
142 		z = Math.floor (parametric_curve.getY (t) + 0.5);
143 		for (y = 0; y < dimy && bufferSize === 0; ++y) {
144 			for (x = 0; x < dimx && bufferSize === 0; ++x) {
145 				connexity = checkVoxel (implicit_curve, x - maxx, y - maxy,
146 					tValues[t]);
147 				if (connexity !== 0){
148 					buffer.push([x, y, z, connexity]);
149 					bufferSize++;
150 					addNeighboursToPile (x, y, t, pile, checked, dimx, dimy,
151 						dimt);
152 				}
153 				checked[x][y][t] = true;
154 			} // end for x
155 		} // end for y
156 	} // end for z
157 	//Incrementation
158 	var voxel;
159 	while (pile.length > 0) {
160 		if (bufferSize >= 1000) {
161 			postMessage ([buffer, bufferSize]);
162 			buffer = [];
163 			bufferSize = 0;
164 		}
165 		voxel = pile.pop ();
166 		checked[voxel[0]][voxel[1]][voxel[2]] = true;
167 		//postMessage(["checkedVoxels"], checked);
168 		if(tValues[voxel[2]] == undefined){
169 			tValues[voxel[2]] = [];
170 			tValues[voxel[2]][0] = parametric_curve.getX([voxel[2]]);
171 			if (tValues[voxel[2]][0] == 0)
172 				tValues[voxel[2]][0] = 0.01;
173 			tValues[voxel[2]][1] = parametric_curve.getX([voxel[2] - 0.5]);
174 			if (tValues[voxel[2]][1] == 0)
175 				tValues[voxel[2]][1] = 0.01;
176 			tValues[voxel[2]][2] = parametric_curve.getX([voxel[2] + 0.5]);
177 			if (tValues[voxel[2]][2] == 0)
178 				tValues[voxel[2]][2] = 0.01;
179 		}
180 		connexity = checkVoxel (implicit_curve, voxel[0] - maxx,
181 			voxel[1] - maxy, tValues[voxel[2]]);
182 		if (connexity !== 0){
183 			buffer.push ([voxel[0], voxel[1], 
184 				Math.floor(parametric_curve.getY (voxel[2]) + 0.5), connexity]);
185 			bufferSize++;
186 			addNeighboursToPile (voxel[0], voxel[1], voxel[2], pile, checked,
187 				dimx, dimy, dimt);
188 		}
189 	}
190 	//post last buffer
191 	postMessage ([buffer, bufferSize,"Terminate"]);
192 }

195 //==============================================================================
196 function init (e) {
197 	parametric_curve = new DrawnCurve ();
198 	parametric_curve.xList = e.data[0][0];
199 	parametric_curve.yList = e.data[0][1];
200 	implicit_curve = new Equation (e.data[1]);
201 	var dimension = new Vector(e.data[2]);
202 	dimx = dimension.x;
203 	dimy = dimension.y;
204 	dimt = parametric_curve.getMaxT();
205 	var x, y, t;
206 	for (x = 0; x < dimx; ++x) {
207 		checked[x] = [];
208 		for (y = 0; y < dimy; ++y) {
209 			checked[x][y] = [];
210 			for (t = 0; t < dimt; ++t) {
211 				checked[x][y][t] = false;
212 			}
213 		}
214 	}
215 	try {
216 		algo ();
217 	}
218 	catch (e) {
219 		console.log (e);
220 		postMessage ([[], 0,"Abort"]);
221 		postMessage ([[], 0,"Terminate"]);
222 	}
223 }

227 /**
228  * Receive the meridian, revolution curve and dimension, then launch algorithm
229  * and ask for termination
230  * 
231  * @param {Event} e - e.data contains the message received.
232  * 
233  * @return {void}
234  */
235 onmessage = function (e) {
236 	if(e.data[0] === "init") {
237 		e.data.splice(0,1);
238 		init(e);
239 	}
240 	else if (e.data[0] === "checkedVoxels") {
241 		checkedVoxels = e.data[1];
242 	}
243 	else if (e.data[0] === "pile") {
244 		pile = e.data[1];
245 	}
246 };
