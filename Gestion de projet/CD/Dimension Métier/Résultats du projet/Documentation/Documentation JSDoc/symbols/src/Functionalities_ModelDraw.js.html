124 //==============================================================================
125 /**
126  * @return {int} The number of points of the curve.
127  */
128 ModelDraw.prototype.getNbPoint = function () {
129 	return this.curve.getMaxT ();
130 };
131 
132 
133 //==============================================================================
134 /**
135  * @return {Number[]} The X coordinates for each point of the curve. 
136  */
137 ModelDraw.prototype.getXCoordinates = function () {
138 	return this.curve.getXList ();
139 };
140 
141 
142 //==============================================================================
143 /**
144  * @return {Number[]} The Y coordinates for each point of the curve. 
145  */
146 ModelDraw.prototype.getYCoordinates = function () {
147 	return this.curve.getYList ();
148 };
149 
150 
151 //==============================================================================
152 /**
153  * Closes the curve. If it is already closed, does nothing.
154  * 
155  * @return {Point} The added point (null if not exist or already added).
156  */
157 ModelDraw.prototype.closeCurve = function () {
158 	var lastIndex = this.getNbPoint () - 1;
159 	
160 	if (lastIndex == -1)
161 		// there isn't any point
162 		return null;
163 		
164 	var lastX = this.getXCoordinates ()[lastIndex],
165 		lastY = this.getYCoordinates ()[lastIndex],
166 		firstX = this.getXCoordinates ()[0],
167 		firstY = this.getYCoordinates ()[0];
168 	
169 	if (lastX != firstX && lastY != firstY) {
170 		this.getActiveCurve ().addPoint (firstX, firstY);
171 		return new Point (firstX, firstY);
172 	}
173 	else 
174 		return null;
175 };
