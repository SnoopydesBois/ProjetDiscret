 76 
 77 /**
 78  * @constructor
 79  */
 80 function ModelParameter () {
 81 
 82 	/**
 83 	 * {Curve} The curve on which the model acts.
 84 	 */
 85 	this.curve = null;
 86 }
 87 
 88 
 89 
 90 //##############################################################################
 91 //	Mutator
 92 //##############################################################################
 93 
 94 
 95 
 96 /**
 97  * Sets the active curve.
 98  * 
 99  * @param {Curve} curve - The curve which parameters are modified.
100  * 
101  * @return {void}
102  */
103 ModelParameter.prototype.setCurve = function (curve) {
104 	this.curve = curve;
105 };
106 
107 
108 
109 //##############################################################################
110 //	Curve modifier
111 //##############################################################################
112 
113 
114 
115 /**
116  * Sets a value to a parameter.
117  * @see {@link Curve.setParameter}
118  * 
119  * @param {String} parameter - The name of the parameter to modify.
120  * @param {Number} value - The value of the parameter to set.
121  * 
122  * @return {void}
123  */
124 ModelParameter.prototype.setParameter = function (parameter, value) {
125 	this.curve.setParameter (parameter, value);
126 };
127 
128 
129 //==============================================================================
130 /**
131  * Returns all the parameters of a curve.
132  * @see {@link Curve.getAllParameters}
133  * 
134  * @return {Map<String, Number>} A map composed of the name of a parameter (the
135  * key) and its value (the value).
136  */
137 ModelParameter.prototype.getAllParameters = function () {
138 	return this.curve.getAllParameters ();
139 };
140 
141 
142 //==============================================================================
143 /**
144  * Retrieves the value of a parameter which name is passed in parameter.
145  * @see {@link Curve.getParameter}
146  * 
147  * @param {String} name - The name of the parameter.
148  * 
149  * @return {Number} The value of the named parameter.
150  */
151 ModelParameter.prototype.getParameter = function (name) {
152 	return this.curve.getParameter (name);
153 };
