 85 /**
 86  * Changes the meridian curve for a model curve. Updates view and parameters
 87  * view.
 88  * @see {@link Controller2D.setActive}
 89  *
 90  * @param {String} name - The name of the new meridian curve.
 91  *
 92  * @return {void}
 93  * @throws {String} Application.changeMeridian: given parameter is not a string
 94  */
 95 Application.prototype.changeMeridian = function (name) {
 96 	/// parameter verification
 97 	if (! checkType (arguments, "string")) {
 98 		throw "Application.changeMeridian: given parameter is not a string";
 99 	}
100 
101 	/// let's change
102 	this.meridianController.setActive (name);
103 	this.meridianView.draw ();
104 
105 	this.meridianParameters.displayParameter (
106 		this.drawMeridian,
107 		this.getRangeMeridian
108 	);
109 };

141 /**
142  * Change the meridian mode (primitive, free hand or formula). Update view and
143  * parameter view.
144  * 
145  * @return {void}
146  * @throws {String} If the mode (i.e. the checked input radio id) is not valid.
147  */
148 Application.prototype.changeMeridianMode = function () {
149 	var mode = $('#meridianType :radio:checked').attr ('id');
150 
151 	/// empty the parameter list
152 	$("#meridianParam").empty ();
153 
154 	if (mode === "meridianPrimitive") {
155 		$("#meridianCanvas").hide (); // hide the div
156 		$("#meridianCanvas2").show (); // display the canvas
157 		var selection = document.getElementById ("listMeridians");
158 		this.meridianController.setActive (selection.value);
159 		/// fill th parameter list
160 		this.meridianParameters.displayParameter (
161 			this.drawMeridian,
162 			this.getRangeMeridian
163 		);
164 		this.exportController.setIdMeridian ("meridianCanvas2");
165 	}
166 	else if (mode === "meridianFreeHand") {
167 		$("#meridianCanvas2").hide ();
168 		$("#meridianCanvas").show ();
169 
170 		this.exportController.setIdMeridian ("meridianCanvas");
171 	}
172 	else if (mode === "meridianFormula") {
173 		$("#meridianCanvas").hide ();
174 		$("#meridianCanvas2").show ();
175 
176 		this.meridianController.setActive (
177 			document.getElementById ("meridianFormulaInput").value);
178 
179 		this.exportController.setIdMeridian ("meridianCanvas2");
180 	}
181 	else {
182 		throw "Application.changeMeridianMode: unkown given mode: " + mode;
183 	}
184 	this.meridianView.draw ();
185 };

189 /**
190  * Changes the revolution mode (primitive or formula). Updates view and parameter
191  * view.
192  * 
193  * @return {void}
194  * @throws {String} If the mode (i.e. the checked input radio id) is not valid.
195  */
196 Application.prototype.changeRevolMode = function () {
197 	var mode = $('#revolType :radio:checked').attr ('id');
198 	$("#revolParam").empty ();
199 	if (mode === "revolPrimitive") {
200 		var selection = document.getElementById ("listRevolutions");
201 		this.revolController.setActive (selection.value);
202 		this.revolutionParameters.displayParameter (
203 			this.drawRevolution,
204 			this.getRangeRevolution
205 		);
206 	}
207 	else if (mode === "revolFormula") {
208 		this.revolController.setActive (
209 			document.getElementById ("revolutionFormulaInput").value,
210 			EquationTypeEnum.IMPLICIT
211 		);
212 	}
213 	else {
214 		throw "Application.changeRevolMode: unkown given mode: " + mode;
215 	}
216 
217 	this.revolView.draw ();
218 };

222 /**
223  * Gets the 3D space size and sets the dimensions of the bounding box and the drawing
224  * meridian canvas.
225  *
226  * @return {void}
227  */
228 Application.prototype.changeDimension = function () {
229 	var dimX = document.getElementById ("dimx");
230 	var dimY = document.getElementById ("dimy");
231 	var dimZ = document.getElementById ("dimz");
232 
233 	var dim = [0, 0, 0];
234 	dim[0] = Math.min (256, Math.max (
235 		parseInt (dimX.value), 1));
236 	dim[1] = Math.min (256, Math.max (
237 		parseInt (dimY.value), 1));
238 	dim[2] = Math.min (256, Math.max (
239 		parseInt (dimZ.value), 1));
240 
241 	dimX.value = dim[0];
242 	dimY.value = dim[1];
243 	dimZ.value = dim[2];
244 
245 	var box = this.surfaceView.getContainer ().getObjectByName ("boundingBox");
246 	box.setDimension (dim);
247 	this.meridianView.draw ();
248 };

292 /**
293  * This function calls itself again every second until the
294  * computation of the surface is finished. Then it redraws the scene.
295  *
296  * @return {void}
297  */
298 Application.prototype.computationFinished = function () {
299 	if (! this.surfaceController.isAlgoFinished ()) {
300 		setTimeout (this.computationFinished.bind (this), 1000);
301 	}
302 	else {
303 		if (this.surfaceController.isAlgoFinished () != "error") {
304 			this.validMessage ("Finished", 0);
305 		}
306 		document.getElementById ("generate1").disabled = false;
307 		document.getElementById ("generate2").disabled = false;
308 		this.stopLoading ();
309 		this.surfaceView.show (true);
310 	}
311 };

315 /**
316  * This function is called by the generate button. Calls the algorithm and draws
317  * the resulting surface.
318  * 
319  * @param {int} mode - Generation algorithm. 0 for incremental and 1 for
320  * brute-force.
321  *
322  * @return {void}
323  */
324 Application.prototype.generateAndDraw = function (mode) {
325 	document.getElementById ("generate1").disabled = true;
326 	document.getElementById ("generate2").disabled = true;
327 	this.showMessage ("Computing...", 0, "#04E");
328 	var dimX = document.getElementById ("dimx").value;
329 	var dimY = document.getElementById ("dimy").value;
330 	var dimZ = document.getElementById ("dimz").value;
331 
332 
333 	this.surfaceController.setDimension ([dimX, dimY, dimZ]);
334 	this.surfaceView.container.getObjectByName ("boundingBox")
335 		.setDimension ([dimX, dimY, dimZ]);
336 	try {
337 		this.surfaceController.generate (mode);
338 	}
339 	catch (e) {
340 		this.alertMessage ("Aborted", 10000);
341 		this.abort ();
342 	}
343 	this.surfaceRenderer = new SurfaceRenderer (
344 		this.surfaceController,
345 		this.surfaceView.getGLContext ()
346 	);
347 	this.surfaceView.container.removeObjectByName (
348 		SurfaceRenderer.getLastSurfaceName ()
349 	);
350 	this.surfaceView.container.addObject (this.surfaceRenderer);
351 
352 	this.computationFinished ();
353 	this.loading ();
354 	this.changeValueSlider ("#slider-rangeX", false, 0, parseInt (dimX));
355 	this.changeValueSlider ("#slider-rangeY", false, 0, parseInt (dimY));
356 	this.changeValueSlider ("#slider-rangeZ", true, 0, parseInt (dimZ));
357 };
