 78 /**
 79  * Computes the points of the curve based on its equation
 80  * 
 81  * @param {Range} ranX - The x-axis range on which to compute the points
 82  * @param {Range} ranX - The y-axis range on which to compute the points
 83  * 
 84  * @return {Point[][]} An array composed of list of points to represent the curve
 85  * (for example, 1/x will create two list of points)
 86  * @throw {String} The equation is not defined or the parameters are not of type Range
 87  */
 88 ExplicitCurve.prototype.computePoints = function (ranX, ranY) {
 89 	if (! ranX instanceof Range || ! ranY instanceof Range) {
 90 		throw "ExplicitCurve.computePoints.ErrorNotARange";
 91 	} else if (!this.equation.check) {
 92 		throw "ExplicitCurve.computePoints.ErrorEquationNotDefined";
 93 	}
 94 	
 95 	var result = [];
 96 	
 97 	var points = [];
 98 	
 99 	var xMin = ranX.getMin();
100 	var xMax = ranX.getMax();
101 	
102 	var step = this.computeStep();
103 	
104 	for(var x = xMin; x <= xMax; x += step){	
105 		var y = this.equation.compute([x]);
106 		
107 		/* 
108 		 * If the point is within the display range,
109 		 * we add that point to the current connex part of the curve
110 		 */
111 		if(ranY.isIn(y)){
112 			points.push(new Point(x,y));
113 		}
114 		/* 
115 		 * Else, we push into the result array the current connex curve computed
116 		 * and we start a new connex part.
117 		 */
118 		else if(points.length > 0){
119 			result.push(points.slice(0));
120 			points = [];
121 		}
122 	}
123 	
124 	/* 
125 	 * Pushing into result the last connex part computed of the curve
126 	 */
127 	if(points.length > 0){
128 		result.push(points.slice(0));
129 	}
130 	return result;
131 
132 };
