114 /**
115  * This function generate the surface using the algorithm for explicit
116  * functions.
117  * 
118  * @return {void}
119  */
120 function algo () {
121 	var maxx = Math.trunc (dimx / 2);
122 	var maxy = Math.trunc (dimy / 2);
123 	var buffer = [];
124 	var bufferSize = 0;
125 	var zValues = [];
126 	var connexity = "";
127 	var x, y, z;
128 	//find first voxel
129 	for (z = 0; z < dimz && bufferSize === 0; ++z) {
130 		zValues[z] = [];
131 		zValues[z][0] = explicit_curve.compute ([z]);
132 		if (zValues[z][0] === 0)
133 			zValues[z][0]=0.001;
134 		zValues[z][1] = explicit_curve.compute ([z - 0.5]);
135 		if (zValues[z][1] === 0)
136 			zValues[z][1]=0.001;
137 		zValues[z][2] = explicit_curve.compute ([z + 0.5]);
138 		if (zValues[z][2] === 0)
139 			zValues[z][2]=0.001;
140 		for (y = 0; y < dimy && bufferSize === 0; ++y){
141 			for (x = 0; x < dimx && bufferSize === 0; ++x){
142 				connexity = checkVoxel (implicit_curve, x - maxx, y - maxy, 
143 					zValues[z]);
144 				if (connexity !== 0) {
145 					buffer.push([x, y, z, connexity]);
146 					bufferSize++;
147 					addNeighboursToPile (x, y, z, pile, checked, dimx, dimy,
148 						dimz);
149 				}
150 				checked[x][y][z] = true;
151 			} // end for x
152 		} // end for y
153 	} // end for z
154 
155 	//Incrementation
156 	var voxel;
157 	while (pile.length > 0) {
158 		if (bufferSize >= 1000) {
159 			postMessage([buffer, bufferSize]);
160 			buffer = [];
161 			bufferSize = 0;
162 		}
163 		voxel = pile.pop ();
164 		checked[voxel[0]][voxel[1]][voxel[2]] = true;
165 		//postMessage(["checkedVoxels"], checked);
166 		if(zValues[voxel[2]] == undefined){
167 			zValues[voxel[2]] = [];
168 			zValues[voxel[2]][0] = explicit_curve.compute([voxel[2]]);
169 			if (zValues[voxel[2]][0] == 0)
170 				zValues[voxel[2]][0]=0.01;
171 			zValues[voxel[2]][1] = explicit_curve.compute([voxel[2] - 0.5]);
172 			if (zValues[voxel[2]][1] == 0)
173 				zValues[voxel[2]][1]=0.01;
174 			zValues[voxel[2]][2] = explicit_curve.compute([voxel[2] + 0.5]);
175 			if (zValues[voxel[2]][2] == 0)
176 				zValues[voxel[2]][2]=0.01;
177 		}
178 		connexity = checkVoxel (implicit_curve, voxel[0] - maxx,
179 			voxel[1] - maxy, zValues[voxel[2]]);
180 		if (connexity !== 0) {
181 			buffer.push([voxel[0], voxel[1], voxel[2], connexity]);
182 			bufferSize++;
183 			addNeighboursToPile(voxel[0], voxel[1], voxel[2], pile, checked,
184 				dimx, dimy, dimz);
185 		}
186 	}
187 	//post last buffer
188 	postMessage ([buffer, bufferSize,"Terminate"]);
189 }

192 //==============================================================================
193 function init (e) {
194 	explicit_curve = new Equation (e.data[0]);
195 	implicit_curve = new Equation (e.data[1]);
196 	var dimension = new Vector (e.data[2]);
197 	dimx = dimension.x;
198 	dimy = dimension.y;
199 	dimz = dimension.z;
200 	var x, y, z;
201 	for (x = 0; x < dimx; ++x) {
202 		checked[x] = [];
203 		for (y = 0; y < dimy; ++y) {
204 			checked[x][y] = [];
205 			for (z = 0; z < dimz; ++z) {
206 				checked[x][y][z] = false;
207 			}
208 		}
209 	}
210 	try {
211 		algo ();
212 	}
213 	catch (e) {
214 		postMessage ([[], 0,"Abort"]);
215 		postMessage ([[], 0,"Terminate"]);
216 	}
217 }

221 /**
222  * Receive the meridian, revolution curve and dimension, then launch algorithm
223  * and ask for termination
224  * 
225  * @param {Event} e - e.data contains the message received.
226  * 
227  * @return {void}
228  */
229 onmessage = function (e) {
230 	if (e.data[0] === "init") {
231 		e.data.splice (0,1);
232 		init (e);
233 	}
234 	else if (e.data[0] === "checkedVoxels") {
235 		checkedVoxels = e.data[1];
236 	}
237 	else if (e.data[0] === "pile") {
238 		pile = e.data[1];
239 	}
240 };
