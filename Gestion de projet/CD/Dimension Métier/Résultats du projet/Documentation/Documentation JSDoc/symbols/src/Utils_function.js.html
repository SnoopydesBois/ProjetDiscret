 62 /**
 63  * Clamps the number between mini and maxi.
 64  *
 65  * @param {Number} mini - The minimum limit.
 66  * @param {Number} maxi - The maximum limit.
 67  * @param {Number} number - The number to clamp between mini and maxi.
 68  *
 69  * @return {Number} 'number' if it is between 'mini' and 'maxi', else 'maxi' or
 70  * 'mini'.
 71  */
 72 function clamp (mini, maxi, number) {
 73 	return Math.min (maxi, Math.max (mini, number));
 74 };

 78 /**
 79  * Computes the polar angle of a 2D point (in radian).
 80  *
 81  * @param {Number} x - X coordinate.
 82  * @param {Number} y - Y coordinate.
 83  *
 84  * @return {Number} The angle of the vector in radian.
 85  */
 86 function angle (x, y) {
 87 	var len = Math.sqrt (x * x + y * y),
 88 		xNorm = x / len;
 89 	return (y < 0) ? Math.acos (-xNorm) + Math.PI : Math.acos (xNorm);
 90 };

 94 /**
 95  * Computes the azymuthal angle of a 3D point for a given origin.
 96  * 
 97  * @param {Vector} point - A point.
 98  * @param {Vector} origin - The origin.
 99  * 
100  * @return {Number} The azimuthal angle of 'point' spheric coordinate (for the 
101  * specified origin).
102  */
103 function getAzimuth (point, origin) {
104 	var pos = new Vector (point).sub (origin);
105 	return angle (pos.x, pos.y);
106 };

110 /**
111  * Computes the altitude angle of a 3D point for a given origin.
112  * 
113  * @param {Vector} point - A point.
114  * @param {Vector} origin - The origin.
115  * 
116  * @return {Number} The altitude angle of 'point' spheric coordinate (for the 
117  * specified origin).
118  */
119 function getAltitude (point, origin) {
120 	var pos = new Vector (point).sub (origin);
121 	return Math.asin (pos.z / pos.getLength ());
122 };

132 /**
133  * Transforms a face position into a color.
134  *
135  * @param {Voxel} voxel - The voxel.
136  * @param {DirectionEnum} direction - The direction of the face in the voxel.
137  * @param {Vector} universSize - Size of the 3D model.
138  *
139  * @return {float[4]} The RGBA color corresponding to the position of the face.
140  * Each component is a value between 0.0 and 1.0.
141  */
142 function posToColor (voxel, direction, universSize) {
143 	return [
144 		voxel.getPosition().x / universSize.x, // red
145 		voxel.getPosition().y / universSize.y, // green
146 		voxel.getPosition().z / universSize.z, // blue
147 		1.0 - direction / 100 // alpha
148 	];
149 };

153 /**
154  * Transforms a face color into a position.
155  *
156  * @param {float[4]} color - A color.
157  * @param {Vector} universSize - Size of the 3D model.
158  *
159  * @return {Facet} The corresponding facet.
160  */
161 function colorToPos (color, universSize) {
162 	return new Facet (
163 		new Vector (
164 			Math.round (color[0] * universSize.x / 256), // red   -> x
165 			Math.round (color[1] * universSize.y / 256), // green -> y
166 			Math.round (color[2] * universSize.z / 256)  // blue  -> z
167 		),
168 		Math.round ((1.0 - color[3] / 256) * 100) // alpha -> direction
169 	);
170 };

180 /**
181  * Adds an event to an element. Does nothing if the given element is null or
182  * undefined.
183  * 
184  * @param {HTMLElement} elem - The element to which we add the event.
185  * @param {String} type - The kind of event.
186  * @param {Function} eventHandle - The handler of the event.
187  * 
188  * @return {void}
189  */
190 function addEvent (elem, type, eventHandle) {
191 	if (elem == null || typeof (elem) == 'undefined')
192 		return;
193 
194 	if (elem.addEventListener)
195 		elem.addEventListener (type, eventHandle, false);
196 	else if (elem.attachEvent)
197 		elem.attachEvent ("on" + type, eventHandle);
198 	else
199 		elem["on"+type] = eventHandle;
200 };

211 /**
212  * @param {*} obj - The object to test.
213  *
214  * @return {String} The 'obj' class name or type if 'obj' is not an object.
215  */
216 function type (obj) {
217 	switch (obj) {
218 	case undefined :
219 		return "undefined";
220 	case null :
221 		return "null";
222 	default :
223 		return ((typeof obj == "object") || (typeof obj == "function") ?
224 			obj.constructor.name :
225 			typeof obj
226 		);
227 	}
228 };

249 /**
250  * Checks if a variable have the expected type.
251  *
252  * @param {*} variable - An object.
253  * @param {(String | Function)} expectedType - The expected type of argument.
254  *
255  * @return {boolean} True if 'variable' has the expected type, false otherwise.
256  */
257 function isA (variable, expectedType) {
258 	if (expectedType === null)
259 		return variable === null;
260 	else if (expectedType === undefined)
261 		return variable === undefined;
262 	else if (typeof expectedType == "string")
263 		return typeof variable == expectedType;
264 	else
265 		return variable instanceof expectedType;
266 };

270 /**
271  * Checks if a list of variables have the expected type(s). The N variable in
272  * 'args' is tested with N + 1 type in the argument list. If an expected type is provided in an
273  * array, the value is tested with each type in the array.
274  *
275  * @param {Array} args - The list of argument.
276  * @param {...(String | Function | (String | Function)[])} Expected type(s) of
277  * argument.
278  *
279  * @return {boolean} True if every item in 'args' have the expected type, false
280  * otherwise.
281  */
282 function checkType (args) {
283 	var nbTest = Math.min (args.length, arguments.length - 1);
284 
285 	var i = 0;
286 	var goodArgs = true;
287 	while (goodArgs && i < nbTest) {
288 		if (arguments[i + 1] instanceof Array) {
289 			var goodType = false, nbType =  arguments[i + 1].length, j = 0;
290 			while (!goodType && j < nbType) {
291 				goodType = goodType || (isA (args[i], arguments[i + 1][j]));
292 				++j;
293 			}
294 		}
295 		else
296 			goodArgs = goodArgs && (isA (args[i], arguments[i + 1]));
297 		++i;
298 	} // end while
299 	return goodArgs;
300 };

304 /**
305  * Searches if a value is in an enumeration.
306  *
307  * @param {Object} enumeration - An enumeration-like built with an object. Its
308  * strucure is {KEY_1: NUMBER, KEY_N: NUMBER}.
309  * @param {(Number | String)} value - The value to test.
310  */
311 function isValueOfEnum (enumeration, value) {
312 	for (var i in enumeration)
313 		if (value == enumeration[i])
314 			return true;
315 	return false;
316 };
